    /*
        ============================ 型 ============================
        - Inint : デフォルト初期値を定義した long long
    */



    template<long long init_v>
    class Inint{
        private:
        long long data;

        public:
        Inint(long long x):data(x){}
        explicit Inint() : Inint(init_v) {}
        
        operator long long(){return data;}
    };


    /*
        ============================ アルゴリズム ============================
        - findKthSmallest(A,k) := 配列 A の要素のうち、小さいものから k 番目のものを答える。O(|A|) 時間だが sort とあんまり変わらない。
    */


    // 配列 A で、小さいものから k 番目の要素を答える (0-index)
    // 一応 O(|A|) 時間で、ソートよりはすこーし速い程度
    template<typename T>
    T findKthSmallest(vector<T> A , int k){
        static const int blocksize = 15;
        static const int halfsize = (blocksize>>1);
        assert(0 <= k && k < A.size());
        
        if(A.size() <= blocksize){
            sort(A.begin(),A.end());
            return A[k];
        }
        
        // blocksize ちょうどにならないブロックは、最大値で埋めておいても一般性を損なわない
        int edge = blocksize - int(A.size())%blocksize;
        if(edge < blocksize)while(edge-->0)A.push_back(numeric_limits<T>::max());

        vector<T> B;
        for(int b = 0 ; b*blocksize < A.size() ; b++){
            sort(A.begin() + b*blocksize , A.begin() + (b+1)*blocksize);
            B.push_back(A[b*blocksize + halfsize]);
        }

        // 要素が偶数個の中央値の代わりに、中央 2 つのうち index が小さい方を見る (余分に埋めた領域の方の値がヒットしないように)
        T mid = findKthSmallest<T>(B,(int(B.size())-1)>>1);
        int mid_cnt = 0;
        vector<T> S,L;
        for(T x : A){
            if(x < mid)S.push_back(x);
            else if(x == mid)mid_cnt++;
            else L.push_back(x);
        }

        if(k < S.size())return findKthSmallest<T>(S,k);
        if(k < S.size() + mid_cnt)return mid;
        return findKthSmallest<T>(L,k - int(S.size()) - mid_cnt);
    }



    /*
        ============================ 数値計算 ============================
        - intersect(l1,r1,l2,r2) := 二つの区間 [ l1 , r1 ) と [ l2 , r2 ) が交差するかを判定
        - intersection_size(l1,r1,l2,r2) := 二つの区間 [ l1 , r1 ) と [ l2 , r2 ) の共通部分の大きさ
        - dist_on_ring(a,b,c,S) := 円周が S の円環上で地点 a から地点 b まで、地点 c を通らずに移動する距離
        - timecount(y,mo,d,h,mi,s) := 西暦 0/1/1 00:00:00 から西暦 y/mo/d h::mi::s までの秒数
            |- ただしグレゴリオ歴なので、1582 年以前は未定義
        - datecount(y,mo,d) := 西暦 0/1/1 から西暦 y/mo/d まで何日か (何日経過したか)
            |- ただしグレゴリオ歴なので、1582 年以前は未定義
    */

    // 二つの区間 [ l1 , r1 ) と [ l2 , r2 ) が交差するか
    template<typename T>
    bool intersect(T l1 , T r1 , T l2 , T r2){return (l1 <= l2 && l2 < r1) || (l2 <= l1 && l1 < r2);}

    // 二つの区間 [ l1 , r1 ) と [ l2 , r2 ) の共通部分の大きさ
    template<typename T>
    T intersection_size(T l1 , T r1 , T l2 , T r2){
        if(!intersect_range(l1,r1,l2,r2))return 0;
        return min<T>(r1,r2)-max<T>(l1,l2);
    }

    // 円周が S の円環上で地点 a から地点 b まで、地点 c を通らずに移動する距離
    template<typename T>
    T dist_on_ring(T a , T b , T c , T S){
        assert(a!=c && b!=c);// c と重複してしまう場合はエラー
        if(a > b)swap(a,b);// 対称性より swap して OK
        if(a < c && c < b)return a+S-b;
        return b-a;
    }

    // グレゴリオで、西暦 0 年 1 月 1 日 0 時 0 分 0 秒から、西暦 Y 年 Mo 月 D 日 H 時 Mi 分 S 秒までの経過時間を計算
    // ただし、0/1/1 00:00:00 は数値計算用の絶対的な値として採用しているだけで、グレゴリオ歴は 1582 年スタートであることに注意
    long long timecount(long long Y , long long Mo , long long D , long long H = 0 , long long Mi = 0 , long long S = 0){
        // datecount for month [x] := x月の日数。ただし閏年ではない
        static const int d4m[13] = {0,31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        static const int accum_d4m[14] = {0 , 0 , 31 , 59 , 90 , 120 , 151 , 181 , 212 , 243 , 273 , 304 , 334 , 365};// [x] := その年から x 月の初めまでの日数
        // キャスト用に long long にしておく
        static const long long h4d = 24;// 一日あたりの時刻数
        static const long long m4h = 60;// 一時間あたりの分数
        static const long long s4m = 60;// 一分あたりの秒数
        assert(0 <= Y);
        assert(1 <= Mo && Mo <= 12);
        if(Mo == 2 && ((Y%4 == 0 && Y%100 != 0) || Y%400 == 0))assert(1 <= D && D <= d4m[Mo]+1);
        else assert(1 <= D && D <= d4m[Mo]);
        assert(0 <= H && H <= 23);
        assert(0 <= Mi && Mi <= 59);
        assert(0 <= S && S <= 59);

        long long daycount = 0;
        if(Y>=1)daycount += (long long)((Y-1)/400)*366 + ((long long)((Y-1)/100)-(long long)((Y-1)/400))*365 + ((long long)((Y-1)/4)-(long long)((Y-1)/100))*366 + ((Y-1)-(long long)((Y-1)/4))*365;
        daycount += accum_d4m[Mo] + D - 1;
        if(Mo > 2 && ((Y%4 == 0 && Y%100 != 0) || Y%400 == 0))daycount += 1;
        return daycount*h4d*m4h*s4m + H*m4h*s4m + Mi*s4m + S;
    }
    
    // 0/1/1 から　Y/Mo/D までの日数 (!!!Y/Mo/D を含まない!!!)
    // つまり、0/1/1 から何日経過すると Y/Mo/D になるかを計算
    long long datecount(long long Y , long long Mo , long long D){
        return (long long)((timecount(Y,Mo,D)-timecount(0,1,1))/86400) + 1;
    }
