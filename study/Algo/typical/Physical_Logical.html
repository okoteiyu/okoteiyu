template<typename T>
class LgVector{
    private:
    vector<pair<T,long long>> m_data;
    long long m_timer;// 最後に init したタイミング
    long long m_cur;// mod m_data.size()
    bool m_rev;
    T m_init_value;

    public:
    LgVector(int n , T init_v = T()) :m_data(n,{init_v,0}) , 
        m_timer(0),m_init_value(init_v), m_cur(0) , m_rev(false){}


    // 配列全体を init_v で初期化する
    void init(T init_v){
        m_timer++;
        m_init_value = init_v;
    }

    // 全体を反転
    void reverse(){
        m_rev = !m_rev;
        m_cur = (int(m_data.size()) - m_cur)%int(m_data.size());
    }

    // 全体を右に c 回転
    void rotate_right(long long c){
        m_cur = (m_cur - c%(long long)(m_data.size()) + m_data.size())%(long long)(m_data.size());
    }

    // i 番目の要素にアクセス
    T& operator[](int i){
        assert(0 <= i && i < m_data.size());
        long long rot = m_cur;
        i = (i+rot+m_data.size())%int(m_data.size());
        if(m_rev)i = (int(m_data.size()) - i - 1)%int(m_data.size());

        assert(0 <= i && i < m_data.size());
        // init よりの前のデータはアクセス時に破棄する
        if(m_data[i].second != m_timer){
            m_data[i].second = m_timer;
            m_data[i].first = m_init_value;
        }
        return m_data[i].first;
    }
    
    int size(){return this->m_data.size();}

    void Debug(){
        cerr << "rev : " << bool(m_rev) << endl;
        cerr << "rot : " << m_cur << endl;
        cerr << "debug : ";
        for(int i = 0 ; i < size() ; i++)cerr << (*this)[i] << " ";
        cerr << endl;
    }
};
