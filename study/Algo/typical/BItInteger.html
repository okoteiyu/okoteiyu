ビット操作簡単!!

    整数 + ビットへの [] オペレータアクセス(読み書き)
    - 整数 -> BitInt のキャストは暗黙
    - BitInt -> 整数 のキャストは明示
    |           
    - 主に bit 全探索や bit DP で使う。
        |- for 文ループでインクリメントしながら探索するとき
            / インクリメントのコストの影響を減らすために
            / このクラスはなるべく外側のループで回すようにしよう
*/
template<int bitsize = 32>
struct BitInt{
    /*   
        Copyright © (c) okoteiyu 2024. 
        Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
    */

    using bit_ = bool;
    bit_ V[bitsize];

    // IntValue をビット列 V など、IntValue のビットの情報をもつメンバに反映
    void assign(unsigned long long val){
        for(int b = 0 ; b < bitsize ; b++)V[b] = bit_(((1ull<<b) & val) != 0);
    }

    void free(){
        delete [](this->V);
    }
    public:

    int popcount() const& {
        int res = 0;
        for(bit_ b : V)if(b)res++;
        return res;
    }

    unsigned long long toint() const&  {
        unsigned long long res = 0;
        for(int b = 0 ; b < bitsize ; b++)res += (1ull<<b)*(unsigned long long)(V[b]);
        return res;
    }

    BitInt(){this->assign(0);}
    BitInt(unsigned long long val){
        this->assign(val);
    }
    
    // キャスト (普段使う型のみ (ビットさえ正しければ、符号付きとかはあんまり関係ない))
    explicit operator char(){static_assert(bitsize<=8);return this->toint();}
    explicit operator short(){static_assert(bitsize<=16);return this->toint();}
    explicit operator int(){static_assert(bitsize<=32);return this->toint();}
    explicit operator long long(){static_assert(bitsize<=64);return this->toint();}

    
    // 重い
    BitInt& operator +=(const BitInt& x){
        *this = BitInt(this->toint()+x.toint());
        return *this;
    }

    // 重い
    BitInt& operator -=(const BitInt& x){
        *this = BitInt(this->toint()-x.toint());
        return *this;
    }

    // 重い
    BitInt& operator |=(const BitInt& x){
        *this = BitInt(this->toint()|x.toint());
        return *this;
    }

    // 重い
    BitInt& operator &=(const BitInt& x){
        *this = BitInt(this->toint()&x.toint());
        return *this;
    }

    // 重い
    BitInt& operator ^=(const BitInt& x){
        *this = BitInt(this->toint()^x.toint());
        return *this;
    }

    // 前置インクリメント (インクリメント後の参照を返す)
    BitInt& operator ++(){
        for(int b = 0 ; b < bitsize ; b++){
            V[b] = !V[b];
            if(V[b])break;
        }
        return *this;
    }

    // 後置インクリメント (インクリメント後の値を返す)
    // int との二項演算だが、第二引数の int を書かなくても呼び出される(不思議)
    BitInt operator ++(int temp){
        BitInt res = *this;
        ++(*this);
        return res;
    }
    
    // 前置デクリメント (デクリメント後の参照を返す)
    BitInt& operator --(){
        for(int b = 0 ; b < bitsize ; b++){
            V[b] = !V[b];
            if(!V[b])break;
        }
        return *this;
    }

    // 後置デクリメント (デクリメント後の値を返す)
    // int との二項演算だが、第二引数の int を書かなくても呼び出される(不思議)
    BitInt operator --(int temp){
        BitInt res = *this;
        ++(*this);
        return res;
    }

    // (下から) i 番目の bit (2^i) に書き込み可能アクセス
    bit_& operator [](int i){
        assert(0<=i && i < bitsize);
        return V[i];
    }

    // ビット数
    int size(){return bitsize;}

    // 比較演算
    friend bool operator ==(const BitInt& a , const BitInt& b){return a.toint() == b.toint();}
    friend bool operator !=(const BitInt& a , const BitInt& b){return a.toint() != b.toint();}
    friend partial_ordering operator <=>(const BitInt& a , const BitInt& b){return a.toint() <=> b.toint();}

    // 二項演算
    friend BitInt operator +(BitInt a , BitInt b){return (a+=b);}
    friend BitInt operator -(BitInt a , BitInt b){return (a-=b);}
    friend BitInt operator |(BitInt a , BitInt b){return (a|=b);}
    friend BitInt operator &(BitInt a , BitInt b){return (a&=b);}
    friend BitInt operator ^(BitInt a , BitInt b){return (a^=b);}


    // 出力ストリーム
    friend ostream& operator <<(ostream& os , const BitInt& a){os << a.toint();return os;}

    /*   
        Copyright © (c) okoteiyu 2024. 
        Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
    */
};
