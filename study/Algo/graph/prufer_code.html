class prufer_code{
    public:
    vector<vector<int> > G;
    vector<pair<int,int> > edges;
    vector<int> p;
    int V; //頂点数 (最大の頂点番号は V-1)
    prufer_code(int V_){
        V = V_;
        G.resize(V);
        p.resize(V-1);
    }
    void add_edge(int u ,int v){
        G[u].push_back(v);
        G[v].push_back(u);
    }
    void convert(){
        vector<int> parent(V,-1);
        queue<int> bfs;
        bfs.push(V-1);
        vector<int> degree(V,0);
        while(!bfs.empty()){
            int now = bfs.front();
            bfs.pop();
            degree[now] = G[now].size();
            for(int nx : G[now]){
                if(parent[nx] != -1)continue;
                parent[nx] = now;
                bfs.push(nx);
            }
        }
        priority_queue<int , vector<int> , greater<int> > pq;
        for(int u = 0 ; u < V-1 ; u++){
            if(degree[u] == 1)pq.push(u);// V-1以外の、外部頂点
        }
        for(int i = 0 ; i < int(p.size())-1 ; i++){
            int leaf = pq.top();
            pq.pop();
            p[i] = parent[leaf];
            degree[p[i]]--;
            if(degree[p[i]] == 1)pq.push(p[i]);
        }
        p.back() = V-1;
    }
    void debug(){
        for(int x : p)cerr << x << " ";cerr << endl;
    }
    
    void encode(){
        vector<int> forbidden(V,0);// 0 でない限り、葉頂点の復元の候補にならない
        for(int x : p)forbidden[x]++;
        set<int> s;
        
        for(int v = 0 ; v < V ; v++)s.insert(v);
        //復元part( prufer[i] につながる葉頂点を特定していく)
        for(int x : p)s.erase(x); // 候補から除外
        
        for(int x : p){
            assert(s.size() != 0);
            int child = *s.begin();
            edges.emplace_back(x,child);
            s.erase(child);
            forbidden[x]--;
            if(forbidden[x]==0)s.insert(x);
        }
    }
    void debug_edge(){
        for(pair<int,int> e : edges)cerr << e.first << " "<< e.second << endl;
    }
};
void test(){
    prufer_code p(7);
    p.add_edge(0,1);
    p.add_edge(0,2);
    p.add_edge(2,4);
    p.add_edge(4,5);
    p.add_edge(5,3);
    p.add_edge(2,6);
    p.convert();
    p.debug();
    p.encode();
    p.debug_edge();
