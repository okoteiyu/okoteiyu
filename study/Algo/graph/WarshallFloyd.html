ワーシャル・フロイド法







/*
    ワーシャルフロイド法をサポートする有向グラフクラス
    - add_edge(u,v,w) := (u→v) の重み w の有向辺をメモ
    - build() := メモした辺から、全頂点ペア間の最小辺重み和を計算
    - dynamic_add_edge(u,v,w) := build で計算した結果を動的に変更しながら辺を追加する
*/
template<typename type_Edgeweight>
struct WarshallFloyd{
    /*   
        Copyright ©️ (c) okoteiyu 2024. 
        Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
    */

    public:
    struct Edge{
        int from, to; // 両端の頂点番号 (有向)
        type_Edgeweight weight;
        int id;// 辺にも id を割り振る場合がある
        Edge(){}
        Edge(int f,int t,type_Edgeweight w):from(f),to(t),weight(w){}
    };
    protected:
    // 非到達頂点への距離の初期化値 (加算があるため、上限を 4 で割った値)
    const type_Edgeweight UNREACHED = (numeric_limits<type_Edgeweight>::max()>>4);

    int m_V;// 頂点数 (番号は [0 , m_V) の半開区間)
    vector<vector<type_Edgeweight>> m_dist;// [u][v] := u から v に向かう経路の最小辺重み和
    vector<Edge> m_Edges;// 辺集合
    bool m_built = true;// すでに build が呼び出されたかどうか
    public:

    WarshallFloyd(){}
    WarshallFloyd(int V) : m_V(V){
        m_dist.resize(m_V,vector<type_Edgeweight>(m_V,UNREACHED));
        for(int i = 0 ; i < m_V ; i++)m_dist[i][i] = type_Edgeweight(0);
    }

    // 現時点でメモしている辺重みから、全頂点ペアに対する最小重み和を計算。(O(V^3))
    void build(){
        m_built = true;
        for(int k = 0 ; k < m_V ; k++){
            for(int u = 0 ; u < m_V ; u++){
                for(int v = 0 ; v < m_V ; v++){
                    m_dist[u][v] = min(m_dist[u][k]+m_dist[k][v] , m_dist[u][v]);
                }
            }
        }
    }

    // 有向辺 u→v の辺重みをメモ (O(1))
    void add_edge(int u , int v ,type_Edgeweight w){
        assert(0 <= u && u < m_V);
        assert(0 <= v && v < m_V);
        m_built = false;// 再度 build する必要がある
        m_dist[u][v] = min(m_dist[u][v] , w);
        m_Edges.emplace_back(u,v,w);
    }

    // 有向辺 u→v の辺重みを、全頂点ペア間の最小辺重み和にに反映 (O(V^2))
    void dynamic_add_edge(int u , int v ,type_Edgeweight w){
        assert(m_built);// build 済みである必要がある (そうでなければ、add_edge から build する)
        assert(0 <= u && u < m_V);
        assert(0 <= v && v < m_V);
        m_Edges.emplace_back(u,v,w);
        m_dist[u][v] = min(w,m_dist[u][v]);
        for(int s = 0 ; s < m_V ; s++){
            for(int t = 0 ; t < m_V ; t++){
                m_dist[s][t] = min(m_dist[s][u] + w + m_dist[v][t] , m_dist[s][t]);
            }
        }
    }

    // 頂点 s から t までの最短距離 (重み和)。到達不可能なら UNREACHED = -1 を返す
    // 到達不可能ならば -1 を返す。(重み上限以上の値になる場合も -1 を返す)
    type_Edgeweight dist(int s , int t){
        assert(0 <= s && s < m_V);
        assert(0 <= t && t < m_V);
        assert(m_built);
        if(m_dist[s][t] == UNREACHED)return type_Edgeweight(-1);
        return m_dist[s][t];
    }
    
    // 辺集合
    vector<Edge> edges(){return m_edges;}

    /*   
        Copyright ©️ (c) okoteiyu 2024. 
        Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
    */
};
