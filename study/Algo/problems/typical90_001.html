決めうち二分探索(典型 90 - 1)

長さが \(L\) の羊羹に、\(N\) 個の切れ目がある。\(i\) 番目の切れ目の位置は、羊羹の左端から \(A_i\) の地点である。ちょうど \(K\) 個に切り分けるとき、最も短いものの長さを最大化せよ。

解説

最大化は二分探索です。

  • \(F(x) :=\) 最も短い切れ端の長さが \(x\) 以上になるように切り分けることが可能か

上記の判定関数は単調性があり、\(x\) は [OK / NG] の 2 つの領域に二分されます。実際、\(x = 0\) の場合必ず OK ( 羊羹の長さは正なので )、\(x\) が大きくなってくると条件を満たすのが難しくなり、一度 NG になってからはずっと NG となります。 

int main(){
    ll n , l , k;cin >> n >> l >> k;
    vll a(n);CIN(a);
    a.insert(a.begin(),0);
    a.push_back(l);
    ll rig = INF;// ng 領域
    ll lef = 0; // ok 領域
    while(rig-lef>1){
        ll mid = lef + (rig-lef)/2;
        int cnt = 0;
        ll sum = 0;
        ll now = 0;
        FOR(i,1,a.size()){
            if(sum < mid){
                sum += a[i] - a[i-1];
            }
            if(sum >= mid){
                sum=0;
                cnt++;
            }
        }
        if(sum < mid)cnt--;
        if(cnt < k)rig = mid;
        else lef = mid;
    }
    cout << lef << endl;



括弧列(典型 90 - 2)

( と ) からなる長さ \(N\) の括弧列で、括弧が閉じているものを全て列挙してください。

  • \(N \leq 20\)
解説

まず\(N\) が小さいので、長さ \(N\) の括弧列を全列挙します。\(N\) 文字それぞれが ( と ) のどちらの状態かをビット前探索することで、長さ \(N\) の括弧列を全部列挙できます。

このカッコ列が閉じているための必要十分条件は、全ての接頭辞に対して、( の個数が ) の個数以上であることです。これを for 文で前から確かめます。 

int main(){
    int n;cin >> n;
    vector<string> ans;
    for(int S = 0 ; S < (1<<n) ; S++){
        // S のビットが 0 → (
        // S のビットが 1 → )
        bool ok = true;
        int left_bracket_cnt = 0;// ( がどれだけ余裕があるか
        string s = "";
        for(int b = 0 ;b < n ; b++){
            if(S&(1<<b)){
                left_bracket_cnt--;
                s += ')';
            }else{
                left_bracket_cnt++;
                s += '(';
            }
            if(left_bracket_cnt < 0)ok = false;
        }
        if(left_bracket_cnt != 0)ok = false;
        if(ok)ans.push_back(s);
    }
    sort(ALL(ans));
    for(string s:ans)cout << s << endl;
    return 0;




木の直径(典型 90 - 3)

                        
木が与えられるので、2 つの頂点間の距離の最大値 (+1) を求めてください。
                        
・頂点数 ≦ 200000

木の 2 頂点距離 (辺の本数) の最大値は、2 回 DFS(BFS) を行うことで計算することができます。
  • 適当な根 r から DFS を行い、r から全ての頂点への最短距離を求める(1 回目)
  • r から最も遠い頂点を x として、次は x を根として全ての頂点への最短距離を求める(2 回目)
  • x から最も遠い頂点への距離が答えである。

こうして求めた値を、木の直径と言います。
DFS/BFS については、こちらの記事で書いているので参考にしてください。
stack を使う実装によって、DFS の到達順に頂点を列挙することができます。 
int main(){
    int n;cin >> n;
    vector<vector<int> > g(n+1);
    REP(i,n-1){
        ll u , v;cin >> u >> v;
        g[u].push_back(v);
        g[v].push_back(u);
    }
    vector<int> distfrom1(n+1,1e9);
    distfrom1[1] = 0;
    stack<int> st;
    st.push(1);// 適当な根からの距離を計算する
    while(!st.empty()){
        int now = st.top();
        st.pop();
        for(int nx : g[now] ){
            // 親方向を見ないように
            if(distfrom1[nx] > distfrom1[now]){
                distfrom1[nx] = distfrom1[now]+1;
                st.push(nx);
            }
        }
    }
    vector<int> distfromX(n+1,1e9);
    int X;
    int mx = 0;
    FOR(x,1,n+1){
        if(mx < distfrom1[x]){
            X = x;
            mx = distfrom1[x];
        }
    }
    
    int ans = 0;
    distfromX[X] = 0;
    st.push(X);
    while(!st.empty()){
        int now = st.top();
        st.pop();
        for(int nx : g[now] ){
            // 親方向を見ないように
            if(distfromX[nx] > distfromX[now]){
                distfromX[nx] = distfromX[now]+1;
                // 最遠の点が答え
                ans = max(ans , distfromX[nx]);
                st.push(nx);
            }
        }
    }
    cout << ans+1 << endl;
    return 0;





入出力の高速化(典型 90 - 4)

                        
H×W 行列の i 行 j 列 : (i , j) に A_i_j が書かれています。
                        
各格子 (i , j) について、同じ行または同じ列にある要素をの総和を B_i_j とします。
                        
B 求めて出力してください。
                        
・H , W ≦ 2000

各格子について \(B_{i,j}\) を愚直に計算すると \(O((H+W)HW)\) で間に合いません。
ということで、あらかじめ以下の値を前計算しておくことで高速化を図ります。
  • \(\mathrm{mergeY}_j\) := j 列目の要素 (左から j 列目の縦一列) の和
  • \(\mathrm{mergeX}_i\) := i 行目の要素 (上から i 行目の横一列) の和

すると、\(B_{i,j} = \mathrm{mergeX}_i + \mathrm{mergeY}_j - A_{i,j}\) と計算できます。
\(A_{i,j}\) が重複して加算された分を引くのを忘れないようにしてください。

最後に \(B\) の各値を出力するのですが、H×W が最大 4000000 になってしまい、cout などを使うと出力のオーバーヘッドによって TLE してしまいます。
よって、printf などのより高速な出力を使用する必要があります。 
int main(){
    int h , w ;cin >> h >> w;
    vector<vector<int> > a(h,vector<int>(w));
    REP(i,h){
        REP(j,w)cin >> a[i][j];
    }
    vector<int> mergeY(w,0);
    REP(x,w){
        int sum = 0;
        REP(y,h)sum += a[y][x];
        mergeY[x] = sum;
    }
    vector<int> mergeX(h,0);
    REP(y,h){
        int sum = 0;
        REP(x,w)sum += a[y][x];
        mergeX[y] = sum;
    }
    REP(i,h){
        REP(j,w){
            a[i][j] = mergeY[j] + mergeX[i] - a[i][j];
            printf("%d " , a[i][j]);
        }
        printf("\n");
    }
    return 0;





ダブリング (典型 90 - 5)

                        
 K 種類のリテラル 1 ≦ C_1 , C_2 , ... , C_k ≦ 9を使って N 桁の整数を作るとき、
                        
 B で割り切れる整数は何種類作れるか (mod 1000000007)
                        
・N ≦ 1e18
                        
・K ≦ 9 
                        
・B ≦ 1000

\(\mathrm{dp'}_{i,x}\) := 下から \(i\) 桁を \(B\) で割ったあまりが \(x\) である場合の数
のようにすると DP することができますが、\(B\) が 1000 ぐらいあるので行列累乗しても間に合いません。

そこで下から \(i\) 桁ではなく、
\(\mathrm{dp}_{i,x}\) := 下から \(2^i\) 桁を \(B\) で割ったあまりが \(x\) である場合の数
にすることで、ダブリングすることができます。
  • 下から \(2^{i+1}\) 桁までの答えを、下から \(2^i\) 桁を \(B\) で割った余りと、
       \(2^i+1\) 桁から \(2^{i+1}\) 桁を \(B\) で割った余りから計算する
  • 下から \(2^i+1\) 桁から \(2^{i+1}\) 桁を \(B\) で割った余りは、
      下から \(2^i\) 桁を \(B\) で割った余りに \(10^{2^i}\) をかけたものの余りです

これで、下から \(2^i\) 桁を \(B\) で割った余りが \(x\) になる場合の数を計算できました。
求めたいのは、下から \(N\) 桁を \(B\) で割った余りが 0 の場合の数なので、ここから工夫をします。

\(N\) にダブリング結果を適用するときは、\(N\) の立っているビットに注目します。
\(N\) の下のビットから調べていき、\(2^i\) のビットが立っている場合は、\(\mathrm{dp}_i\) を採用することを考えます。

すると、
\(\mathrm{Ndp}_{i,x}\) := \(N\) の下から \(i\) ビットまで見て作った長さの整数のうち、\(B\) で割った余りが \(x\) である場合の数
を計算することになり、答えは以下のようになります。 
int main(){
    using mint = mint1e9_7;
    ll k , b , n;
    cin >> n >> b >> k;
    vll c(k);CIN(c);
    vector<vector<mint> > dp(63,vector<mint>(b,0));
    // dp[i][x] := (下から)先頭から 2^i 桁 mod B が、x であるような場合の数
    dp[0][0] = 0;
    for(int x : c)dp[0][x%b] += 1;
    
    REP(i,62){
        ll ten = modpow(10%b,(1ll<<i) , b);// 計算が重いので前処理
        REP(x,b){// 先頭 2^i 桁を B で割ったあまりが y
            REP(y,b){// その後ろ 2^i 桁を B で割ったあまりが y
                dp[i+1][(x*ten+ y)%b] += dp[i][x]*dp[i][y];
                // x の前に 2^i 桁あるので、その分 10^(2^i) をかける
            }
        }
    }
    vector dp2(63 , vector<mint>(b , 0));
    dp2[0][0] = 1;// [i][x] := n の(下から) i ビット目まで調べて、B で割った値が x になるような場合の数
    ll now = 0;// n の立っているビットのうち、すでに見たビットのみで 10 進数にしたもの
    for(int bit = 0 ; bit < 62 ; bit++){
        if((1ll<<bit)&n){
            ll ten = modpow(10,now,b) ;
            REP(y,b){// 先頭 now 桁を B で割ったあまりが x
                REP(x,b){// その後ろ 2^bit 桁を B で割ったあまり
                    dp2[bit+1][(x*ten+ y)%b] += dp2[bit][y]*dp[bit][x];
                    // x の方は、その前に now 桁あるので、その分 10^now をかける
                }
            }
            now += (1ll<<bit);
        }else{
            // bit が立っていない箇所も、次のビットの遷移のために、前のビットの結果を受け継ぐ(便宜上)
            REP(x,b)dp2[bit+1][x] += dp2[bit][x];
        }
    }
    cout << dp2[62][0] << endl;
    return 0;




辞書順問題典型(典型 90 - 6)

                        
文字列 S の長さ K の(連続とは限らない)部分列のうち、辞書順最小のものを出力してください。

辞書順最小は前から貪欲に決めていくのが定石です。
つまり、答えとなる文字列 : \(\mathrm{ans}\) の \(i\) 文字目として a を使えるのに、あえて b を使う必要はありません。
\(i\) 文字目で決まった文字列の大小関係が、\(i+1\) 文字目以降で逆転することはないからです。

今回の問題では長さ \(K\) の部分列を求めるので、\(\mathrm{ans}\) の \(i\) 文字目に \(S\) のどの文字でも使って良いわけではありません。
例えば、\(\mathrm{ans}\) の先頭の文字に \(S\) の末尾を採用してしまうと、\(\mathrm{ans}\) の 2 文字目以降を作れなくなってしまいます。

結論から言うと、\(\mathrm{ans}\) の長さが \(K\) であるという制約を実現するために
\(\mathrm{ans}_i\) (\(i\) は 0-index) に採用して良いのは、
  • \(L\) := \(\mathrm{ans}_{i-1}\) に採用した文字の \(S\) 上での \(\mathrm{index} + 1\) (ただし \(i\)=0 の時は \(L\)=0)
  • \(R\) := \(|S| - |\mathrm{ans}| + i\)
としたときに、\(S\) の \(L\) 文字目から \(R\) 文字目までの文字です。(\(L,R\) も 0-index)

\(\mathrm{ans}_i\) として採用する価値があるのは、先述の理由より、選択肢のうち最小の文字のみです。
  • \(\mathrm{ans}_i\) には \(S\) の \(L\) 文字目から \(R\) 文字目までの文字で最小の文字 : \(\mathrm{mc}\) を採用
  • \(\mathrm{ans}_{i+1}\) に採用できる文字の候補を増やすために、
      \(S\) の \(L\) 文字目から \(R\) 文字目までで最も左の \(\mathrm{mc}\) を採用する。
これはセグメントツリーのようなデータ構造によって \(\mathrm{ans}_i\) に採用する値を取得することができますが、
\(S\) のアルファベットサイズが小さい(26)ので、\(\mathrm{ans}_i\) に採用する文字を全部確かめる実装をしてみます。
a ~ z の各文字について \(S\) の何文字目に現れるかを \(\mathrm{set}\) などで昇順に保管しておきます。
\(\mathrm{ans}_i\) に採用する文字を以下の手順で調べます。
  • \(\mathrm{ans}_i\) に採用する文字を a ~ z の順に調べる
  • 今調べている文字を \(c\) とし、
    set に存在する index のうち \(L\) 未満は破棄する(set の先頭から見る)
    残った要素に \(R\) 以下が存在するなら \(c\) を採用する(\(\mathrm{set}\) の先頭を見る)
      (採用した場合、\(c\) 以降は \(\mathrm{ans}_i\) になり得ません。)

以下、解答コードです。 
int main(){
    int n , k;cin >> n >>k;
    string s;cin >> s;
    int l = 0;
    int r = (n-k+1);
    // 半開区間 [l,r) に次に選べる文字の候補がある
    // [c] := 文字 c (0~25 に変換) が現れる S の++使用可能な index ++を昇順に管理
    vector<MyMultiSet<int> > Candidates(26);
    REP(i,n)Candidates[s[i]-'a'].insert(i);
    // 答えの文字列
    string ans = "";
    REP(i,k){
        // ans の i 文字目を何にするかを決める
        REP(c,26){// i 文字目の候補を小さい文字から調べて、使用可能なら採用(それが最適)
            if(Candidates[c].size() == 0)continue;
            // c が現れる使用可能な候補のうち、最左のものが使用可能 (r 未満) なら
            // それを採用するのが最適 (次の文字を選べる範囲を広げたいから)
            if(Candidates[c].get(0).Key < r){
                ans.push_back(c+'a');
                l = Candidates[c].get(0).Key+1;
                r++;
                Candidates[c].Delete(0);
                break;
            }
        }
        // l が変わったので、これまでの候補のうち l よりも左にあるものを削除する
        REP(c,26){
            while(Candidates[c].size() > 0 && Candidates[c].get(0).Key <l)Candidates[c].Delete(0);
        }
    }
    cout << ans << endl;
    return 0;




偏角ソート(典型 90 - 9)

                        
N 個の座標が与えられるので、それらから 3 点 A , B , C を選ぶとき、
                        
ABC のなす角の最大値を求めてください。
                        
・ N ≦ 2000
真ん中の点 B を固定して、残りを偏角ソートします。
その後、各点について、B を中心に 180° 反対側の方角に最も角度が近い点を lower_bound などで調べます。
AngleSorter は平衡二分木に点を乗せ、偏角でソートしたものです。 
int main(){
    ll n;cin >> n;
    vector<Point> P(n);
    REP(i,n){
        ll x , y;cin >> x >> y;
        P[i] = Point(x,y);
    }
    double res = 0;
    REP(i,n){
        DynamicAngleSorter<ll> S(P[i].real() , P[i].imag());
        REP(j,n){
            if(i!=j)S.add_point(P[j].real() , P[j].imag() , j);
        }
        REP(j,n){
            if(j == i)continue;
            // P[i] に対して P[j] と対称な点と角度が近い点を求める
            int it = S.lower_bound(S.Center.x - (P[j].real() - S.Center.x) , S.Center.y - (P[j].imag() - S.Center.y));
            for(int k : { (it-1+S.size())%int(S.size()) , it }){
                if(k < 0 || k >= S.size())continue;
                Point Q(S[k].x,S[k].y);
                if(S.get(k).id == j)continue;
                double th = get_angle(Q,P[i],P[j]);
                while(th < 0)th +=2*PI;
                res = max(res , th*180.0/PI);
            }
        }
    }
    cout.precision(20);
    cout << res << endl;
    return 0;
