MyMultiSet

Set にする。

詳しくは Splay Tree の記事を見てね 






/*
    Copyright © (c) okoteiyu 2024. 
    Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
                   
    - ノードが type_key 型のKey と type_value 型のValueを持つ Splay Tree で、(Key , Value) が +++ 辞書順 +++ に並んでいる
        |- C++ の set とは異なり、Key の重複を許す。(Value も当然重複 OK)
    - get(i) や [i] で i 番目のノードのコピーを取得 (隣接頂点へのアクセス (ポインタ) が封印されたものを返す) (0-index)
    - GetRange(l,r) は半開区間 [l,r) をカバーする部分木の根のコピーを返す。get 同様に、隣接頂点のポインタは封印されている。
        |- GetRange(l,r).Sum_val のようにして [l,r) の持つ要素のモノイド積を取得する
    - (Key,Value) に関して以下の操作が可能
        |- insert_pair(k,v) で (k,v) を持つノードを追加し、erase_pair(k,v) で (k,v) を持つノードを(存在すれば)削除する。
        |- upper_bound_pair(k,v) : (Key,Value) が辞書順で (k,v) 以下の要素数を返す
        |- lower_bound_pair(k,v) : (Key,Value) が辞書順で (k,v) 未満の要素数を返す
        |- count_pair(k,v) : (Key,Value) が (k,v) である要素の個数
        |- find_pair(k,v) : (Key,Value) が (k,v) である要素の index を返す。存在しなければ -1 を返す。(0-index)
    - ノードの Key だけに注目して、通常の set のように振る舞わせることもできる。  
        |- insert(k) : Key = k である要素を追加する。ただし、Value を指定しないので Value は未定義とする。
        |- erase(k) : Key = k である要素を一つ削除する。ただし、Value について特に指定しないことに注意。
        |- upper_bound(k) : Key が k 以下の要素数を返す
        |- lower_bound(k) : Key が k 未満の要素数を返す
        |- count(k) : Key が k である要素の個数
        |- find(k) : Key が k である要素の index を返す。存在しなければ -1 を返す。(0-index)
    - ペアを持つノードとKeyしか持たないノードが混在するといけないので、insert(k) について以下の仕様とする。
        |- insert() を呼び出した時点で、Value が未定義の要素が存在することになるので
            // upper_bound_pair , find_pair などの
            // Value に関する関数を呼び出すとランタイムエラーになるようにしました。
            // m_paired というメンバで管理してます
        |- Key だけに関係する関数は変わらず使用できる。
    - SplayNode の < 演算をを変更することで要素の並び順を変更できる。
    - Splay Tree ライブラリの仕様&設計 : 
        |- 関数に渡す SplayNode や、関数から返される SplayNode は全て update (& eval) されている必要がある
        |- 平衡二分木は、アクセスするだけでも構造を変えるので、const 修飾できない。
        |- get(i) や getRange(l,r) で取得するノードは、要素取得の用途での使用のみを想定しているので、
            / 隣接するノードへのアクセスを封印したコピーを返すようにしてある。
    - 使う際の注意点
        |- 遅延評価を載せる場合は色々注意
            |- 変更後の値が sorted になるような制限をつけないとダメ
*/
template<class type_key , class type_value>
class MyMultiSet{
    /*   
        Copyright © (c) okoteiyu 2024. 
        Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
    */

    private:
    struct SplayNode{
        friend class MyMultiSet;
        
        public:
        
        type_key Key;       // 値 (この値をもとに昇順に並んでいる)
        type_key Sum_key;   // 部分木のノードの Key の値の和

        type_value Value;   // Key とは別に頂点に持たせる値
        type_value Sum_val; // 部分木のノードの val の値の和
        type_value Max_val; // 部分木のノードの val の値の Max
        type_value Min_val; // 部分木のノードの val の値の Min
        
        int SubTreeSize = 1;// 部分木サイズ

        private:
        
        SplayNode *parent = nullptr;  // 親ノード
        SplayNode *left = nullptr;    // 左の子ノード
        SplayNode *right = nullptr;   // 右の子ノード
            
        // ノードのコピーかどうか ( splay tree の get などに詳細 )
        bool temp_instance = false;

        
        SplayNode(){}
        SplayNode(type_key key_ , type_value val_) : Key(key_), Value(val_){
            update();
        }
        
        // デフォルトのシャローコピーを採用
        SplayNode(const SplayNode&) = default;
        SplayNode(SplayNode&&) = default;
        SplayNode& operator=(const SplayNode&) = default;
        SplayNode& operator=(SplayNode&&) = default;


        // データ読み込み用の、隣接頂点へのアクセスを禁止したコピーを返す。
        SplayNode make_temp_instance(){
            assert(temp_instance == false);
            SplayNode res = *this;
            res.left = nullptr;
            res.right = nullptr;
            res.parent = nullptr;
            res.temp_instance = true;
            return res;
        }
        
        
        // this - this->parent を回転させる (親は必ず存在するものとし、位置関係も誤りがないものとする)
        void rotate() noexcept {
            //GrandParentの子を張り替え
            if(this->parent->parent != nullptr) [[likely]] {
                if(this->parent == this->parent->parent->left)this->parent->parent->left = this;
                else this->parent->parent->right = this; 
            }

            this->parent->eval();
            this->eval();
            
            //元の親に対する位置で場合分け
            if(this->parent->left == this){
                this->parent->left = this->right;
                if(this->right != nullptr)this->right->parent = this->parent;
                this->right = this->parent;
                this->parent = this->right->parent;
                this->right->parent = this;
                this->right->update();
            }else{
                this->parent->right = this->left;
                if(this->left != nullptr)this->left->parent = this->parent;
                this->left = this->parent;
                this->parent = this->left->parent;
                this->left->parent = this;
                this->left->update();
            }

            this->update();
            return;
        }

        /*
            自分から見た親の位置
            0 -> 親頂点がない
            1 -> 親の左の場合
            2 -> 親の右の場合
        */
        int state() noexcept {
            if(this->parent == nullptr) [[unlikely]] return 0;
            this->parent->eval();
            if(this->parent->left == this)return 1;
            return 2;
        }

        // あるNodeを回転を駆使し一番上まで持っていく
        void splay() noexcept {
            while(this->parent != nullptr){
                if(this->parent->state() == 0){
                    this->rotate();
                    break;
                }
                if( this->parent->state() == this->state() )this->parent->rotate();
                else this->rotate();
                this->rotate();
            }
            this->update();// while が呼ばれなかった場合のため
            return;
        }

 
        // サイズなどの情報をupdateする
        void update() noexcept {
            assert(temp_instance == false);
            
            this->eval();
            this->SubTreeSize = 1;
            this->Sum_key = this->Key;
            
            this->Sum_val = this->Value;
            this->Max_val = this->Value;
            this->Min_val = this->Value;
            
            if(this->left != nullptr) [[likely]] {
                this->left->eval();
                this->SubTreeSize += this->left->SubTreeSize;
                if(this->left->Min_val < this->Min_val)this->Min_val = this->left->Min_val;
                if(this->left->Max_val > this->Max_val)this->Max_val = this->left->Max_val;
                this->Sum_key += this->left->Sum_key;
                this->Sum_val += this->left->Sum_val;
            }
 
            if(this->right != nullptr) [[likely]] {
                this->right->eval();
                this->SubTreeSize += this->right->SubTreeSize;
                if(this->right->Min_val < this->Min_val)this->Min_val = this->right->Min_val;
                if(this->right->Max_val > this->Max_val)this->Max_val = this->right->Max_val;
                this->Sum_key += this->right->Sum_key;
                this->Sum_val += this->right->Sum_val;
            }
 
            return;
        }

        // 遅延評価を評価する (基本は不要なので不要なので空にしておく)
        void eval() noexcept {
            assert(temp_instance == false);
        }

        // ノードの値によって順序を定義。ただし、共に遅延評価がなく、正しい情報を持っている必要がある。
        // {Key,Value} の辞書順
        friend bool operator<(const SplayNode& a , const SplayNode& b) noexcept {
            if(a.Key != b.Key)return bool(a.Key < b.Key);
            return bool(a.Value < b.Value);
        }

        
    };

    
    protected:
    /*=================================
                メンバ変数たち
    =================================*/
                
    
    // 管理する木の根ノード
    SplayNode *m_Root = nullptr;
    // upper_bound などで比較を行うためのノードオブジェクト。SplayTree に挿入はしない
    SplayNode* const m_bluff_object = new SplayNode();
    // (Key,Value) の両方が有効な値かどうか。insert(k) を呼び出すと未定義の Value が含まれることになるので false になる。
    bool m_paired = true;
    // Value を指定しない場合のデフォルト値
    inline static const type_value m_default_val = type_value(0);
    
    private:
    
    // rootを根とする部分木において、求めたいindexまで降りていく(左側の部分木サイズを参照する)
    static SplayNode *get_sub(int index, SplayNode* root){
        if(root == nullptr)return root;
        assert(root->parent == nullptr);

        auto now = root;
        while(true){
            now->eval();// ノードを見る前にeval
            int left_size = bool(now->left == nullptr) ? 0 : now->left->SubTreeSize;
            if(index == left_size)break;
            if(index < left_size){
                now = now->left;
            }else{
                now = now->right;
                index -= left_size+1;
            }
        }
        now->splay();
        return now;
    }
 

    /*==============================================================================
        merge/split ベースの実装を行う。
        |- merge(lef,rig) : rig の左の子に lef をつける
        |- split(i,rt) : i 番目のノードの左の辺を無くし、(lef,rig) のペアを返す
        merge と split が反対の操作であることを利用して計算量を削減する
    ==============================================================================*/

    
    // leftroot を根とする SplayTree と、rightroot を根とする SplayTree をこの順に結合する。
    static SplayNode *merge(SplayNode *leftRoot, SplayNode *rightRoot){
        if(leftRoot == nullptr)return rightRoot;// 右のみ
        if(rightRoot == nullptr)return leftRoot;// 左のみ
        assert(leftRoot->parent == nullptr && rightRoot->parent == nullptr);
        // 怖いのでここで update かけときます。大したコストではないです 
        leftRoot->update();
        rightRoot->update();

        rightRoot = get_sub(0,rightRoot);
        rightRoot->left = leftRoot;
        leftRoot->parent = rightRoot;
        rightRoot->update();
        return rightRoot;
    }
    
    // root が根の SplayTree を、左から i-1 番目と i 番目の間で分割する。
    std::pair<SplayNode*,SplayNode*> split(int i, SplayNode*root){
        if(root == nullptr)return std::make_pair(nullptr , nullptr);
        assert(root->parent == nullptr);
        assert(0 <= i && i <= root->SubTreeSize);
        if(i == 0)return std::make_pair(nullptr , root);
        if(i == root->SubTreeSize)return std::make_pair(root, nullptr);
        root = get_sub(i , root);
        auto leftRoot = root->left;
        auto rightRoot = root;
        if(rightRoot != nullptr)rightRoot->left = nullptr;
        if(leftRoot != nullptr)leftRoot->parent = nullptr;
        leftRoot->update();
        rightRoot->update();
        return std::make_pair(leftRoot,rightRoot);
    }
    

    /*
        root が根であるような Splay 木で、以下の条件を満たすノード x の個数を返す。
        lower = true : x < node (lower_bound)
        lower = false: x <= node (upper_bound)
        条件を満たすギリギリの境界を探索する (ただし、境界の左右どちらかは不定)
    */
    static std::pair<SplayNode*,int> bound_sub(SplayNode* Node, SplayNode *root , bool lower){
        if(root == nullptr)return std::make_pair(root,0);
        SplayNode *now = root;// 条件による境界ギリギリのノードを根から探索する
        Node->update();// 念の為集約と遅延評価をしておく
        bool satisfy = false;
        while(true){
            now->eval();
            if(lower)satisfy = bool(*now < *Node); // lower_bound (now < Node)
            else satisfy = !bool(*Node < *now);    // upper_bound (now <= Node)
            // now が条件を満たす側にいる場合、右側に境界があるとわかる
            if(satisfy){
                if(now->right == nullptr)break;
                now = now->right;
            }else{
                if(now->left == nullptr)break;
                now = now->left;
            }
        }
        int res = 0;
        // now が条件を満たす側で終了した場合、自身の分を加算する。
        if(satisfy)res = 1;
        now->splay();
        if(now->left != nullptr)res += now->left->SubTreeSize;
        return std::make_pair(now ,res);
    }

    
    // root が根の splaytree に NODE (別の連結成分)を挿入する
    // NODE は update されている必要がある
    SplayNode *insert_sub(SplayNode *NODE , SplayNode *root){
        NODE->update();// 念の為集約と遅延評価をしておく
        if(root == nullptr)return NODE;
        
        // lower_bound で、x < NODE の境界となる x を取得 (ただし境界の右側か左側かは不定)
        root = bound_sub(NODE,root,true).first;
        // root が x < NODE の境界の左右どちらかによって、NODE を配置する場所が変わる
        root->eval();
        if(*root < *NODE){ // root < NODE の場合は root の右に
            if(root->right != nullptr)root->right->parent = NODE;
            NODE->right = root->right;
            root->right = nullptr;
            NODE->left = root;
        }else{// NODE <= root の場合は root の左に
            if(root->left != nullptr)root->left->parent = NODE;
            NODE->left = root->left;
            root->left = nullptr;
            NODE->right = root;
        }
        root->parent = NODE;
        root->update();
        NODE->update();
        return NODE;// NODE が根になっている
    }
     

    // upper_bound などで比較を行うためのノードオブジェクト。SplayTree に挿入はしない
    SplayNode* const BluffObject(type_key k , type_value v){
        m_bluff_object->Key = k;
        m_bluff_object->Value = v;
        return m_bluff_object;
    }

    public:

    MyMultiSet(): m_Root(nullptr) , m_paired(true){}
    ~MyMultiSet(){
        // メモリ解放。MyBinaryVector のような要素検索がある場合は、多重解放にならないようにしてね
        while(m_Root != nullptr)this->pop(0);
    }

    // コピー
    MyMultiSet(const MyMultiSet<type_key,type_value>& param_tree) : MyMultiSet() {
        // subr 以降の部分木をコピー
        function<SplayNode*(const SplayNode*)>subr_copy = [&](const SplayNode* subr){
            if(subr == nullptr)return static_cast<SplayNode*>(nullptr);
            SplayNode* res = new SplayNode(*subr);
            res->left = subr_copy(res->left);
            res->right = subr_copy(res->right);
            if(res->left != nullptr)res->left->parent = res;
            if(res->right != nullptr)res->right->parent = res;
            return res;
        };
        m_Root = subr_copy(param_tree.m_Root);
        this->m_paired = param_tree.m_paired;
    }
    
    // ムーブ
    MyMultiSet(MyMultiSet<type_key,type_value>&& param_tree) noexcept : MyMultiSet() {
        swap(this->m_Root , param_tree.m_Root);
        swap(this->m_paired , param_tree.m_paired);
    }
    
    // コピー&スワップ イディオム
    MyMultiSet& operator=(MyMultiSet<type_key,type_value> param_tree){
        swap(this->m_Root , param_tree.m_Root);
        swap(this->m_paired , param_tree.m_paired);
        // param_tree に渡したリソースは param_tree のデストラクタで破棄される
        return *this;
    }

    // 要素数
    int size(){
        if(m_Root == nullptr)return 0;
        return m_Root->SubTreeSize;
    }

    // 左から i 番目のNodeのコピーを取得 (0-index)
    // ただし、値の取得が目的なので right などの隣接頂点へのアクセスを封印する
    SplayNode get(int i){
        assert(0 <= i && i < size());
        m_Root = get_sub(i,m_Root);
        return m_Root->make_temp_instance();
    }


    // [l,r) に対応する部分木の根のコピーを返す。ただし、隣接ノードへのアクセスを封じる。
    // 使用用途はモノイド積の取得に限定 (コピーなので、オリジナルの現在の値とは関係なく [l,r) のデータが格納されている)
    SplayNode GetRange(int l , int r){
        assert(0 <= l && l < r && r <= size());
        auto tmp = split(r,m_Root);
        auto rightRoot = tmp.second;
        tmp = split(l,tmp.first);// 部分木を取り出す。
        auto res = tmp.second->make_temp_instance();
        m_Root = merge(merge(tmp.first,tmp.second),rightRoot);
        return res;
    }
 
    // Keyのみを指定して挿入 ( Value は指定しないので未定義 )
    void insert( type_key key_ ){
        // 未定義の Value が追加されたので、今後 (Key,Value) の pair に注目する操作が行えなくなる
        m_paired = false;
        m_Root = insert_sub(new SplayNode(key_,m_default_val) ,m_Root);
        return;
    }
 
    // KeyとValueを指定して挿入
    void insert_pair( type_key key_ , type_value val_){
        assert(m_paired);
        m_Root = insert_sub(new SplayNode(key_,val_) ,m_Root);
        return;
    }
 
    // index番目のNodeを消去
    void pop(int index){
        assert(0 <= index && index < size());
        auto center = get_sub(index,m_Root);
        auto leftRoot = center->left;
        auto rightRoot = center->right;
        if(leftRoot != nullptr)leftRoot->parent = nullptr;
        if(rightRoot != nullptr)rightRoot->parent = nullptr;
        center->left = nullptr;
        center->right = nullptr;
        center->update();

        m_Root = merge(leftRoot,rightRoot);
        delete center;
    }

    // Keyとして、key_を持つ頂点を一つpop。ただし、Valueについては指定しない
    void erase(type_key key_){
        int it = find(key_);
        if(it!=-1)pop(it);
        return;
    }
 
    /*(key_,val_)を持つ頂点を一つ削除*/
    void erase_pair(type_key key_ , type_value val_){
        assert(m_paired);
        int it = find_pair(key_ , val_);
        if(it!=-1)pop(it);
        return;
    }
 
    // Key が x 未満の頂点の個数を返す    
    int lower_bound(type_key x){
        if(size() == 0)return 0;
        // 比較用 Node オブジェクトを使って比較する
        auto tmp = bound_sub(BluffObject(x,type_value(0)),m_Root,true);
        m_Root = tmp.first;
        return tmp.second;
    }

    // (Key , Value)に関して、a < NODE(x,y)であるような頂点の個数を返す
    int lower_bound_pair(type_key x , type_value y){
        assert(m_paired);
        if(size() == 0)return 0;
        auto tmp = bound_sub(BluffObject(x,y),m_Root,true);
        m_Root = tmp.first;
        return tmp.second;
    }
 
    // Key が x 以下の頂点の個数を返す
    int upper_bound(type_key x){
        if(size() == 0)return 0;
        auto tmp = bound_sub(BluffObject(x,type_value(0)),m_Root,false);
        m_Root = tmp.first;
        return tmp.second;
    }
 
    // (Key , Value)に関して、a <= NODE(x,y)であるような頂点の個数を返す
    int upper_bound_pair(type_key x , type_value y){
        assert(m_paired);
        if(size() == 0)return 0;
        auto tmp = bound_sub(BluffObject(x,y),m_Root,false);
        m_Root = tmp.first;
        return tmp.second;
    }
    

    // x の個数を返す
    int count(type_key x){
        return upper_bound(x) - lower_bound(x);
    }

    // (x,y) の個数を返す
    int count_pair(type_key x , type_key y){
        assert(m_paired);
        return upper_bound_pair(x,y) - lower_bound_pair(x,y);
    }

    
    // Key == x である index (複数あるなら最初のもの) を返す (0-index)。なければ-1を返す
    int find(type_key x){
        if(size() == 0)return -1;
        if(count(x) == 0)return -1;
        return lower_bound(x);
    }

    // (Key,Value) == (x,y) である index (複数あるなら最初のもの) を返す (0-index)。なければ-1を返す
    int find_pair(type_key x , type_value y){
        assert(m_paired);
        if(size() == 0)return -1;
        if(count_pair(x,y) == 0)return -1;
        return lower_bound_pair(x,y);
    }


    // 一列に表示
    void Debug(){
        std::cerr<<"DEBUG: -- Size = " << size() << std::endl;
        for( int i = 0 ; i < size() ; i++)std::cerr<< get(i).Key << " ";
        std::cerr<< std::endl;
        if(m_paired == false)return ; 
        for( int i = 0 ; i < size() ; i++)std::cerr<< get(i).Value << " ";
        std::cerr<< std::endl;
    }
 
    // 末尾を取得
    SplayNode back(){
        assert(size()>0);
        return get(size()-1);
    }
    // 先頭を取得
    SplayNode front(){
        assert(size()>0);
        return get(0);
    }
    // 末尾を削除
    void pop_back(){
        assert(size()>0);
        pop(size()-1);
    }
    // 先頭を削除
    void pop_front(){
        assert(size()>0);
        pop(0);
    }
    // read-only でアクセス
    SplayNode operator [](int index){
        return get(index);
    }
