    /*
        MOD つき整数。自動で mod をとってくれる。
        基本的な演算は整数型と同じような設計にしてあり、整数型との演算も行える。
        mod が素数の場合、0 以外の整数に逆数が存在する。  
            |- 「%」演算は意味をなさない (必ず 0)
        - 設計
            |- 整数型 -> modint のキャストは暗黙
                |- これにより、modint 同士の演算のみ定義しておけば、整数型との演算もカバーできる
            |- modint -> 整数型 のキャストは明示的
                |- 参照する演算が複数存在するような状況を回避するため。
    */
    template<long long MOD , typename = enable_if_t<bool(0 < MOD && MOD <= 2'000'000'000)>>
    struct modint{
        /*   
            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
        protected:
        long long m_value = 0;

        public:
        modint(){}

        // long long からの変換コンストラクタ
        modint(long long v_) : m_value(v_ % MOD){
            if(m_value < 0)m_value += MOD;
        }
        
        modint(const modint& t) : modint(t.m_value) {}
        modint(modint&& t) : modint(t.m_value) {}
            
        
        modint& operator=(modint t){
            this->m_value = t.m_value;
            return *this;
        }


        // 値を取り出す
        long long val() const& {
            return m_value;
        }

        // 自身の b 乗
        modint pow(long long b) const& {
            modint a = (*this);
            modint res = 1;
            modint acum = a;
            while(b > 0){
                if( b & (1ll) )res *= acum;
                acum *= acum;
                b >>= 1;
            }
            return res;
        }

        // 自身の逆数
        modint inv() const& {
            assert(this->m_value % MOD != 0);
            modint b_ = (*this);
            long long b = b_.m_value;
            long long x = MOD;
            modint u = modint(1);
            modint v = modint(0);
            while (x!=0) {
                long long t = b / x;
                b -= t * x; 
                swap(b, x);
                u -= t * v; 
                swap(u, v);
            }
            assert(u*b_ == 1);
            return u;
        }

        // - 符号をつけた値
        modint operator-() const& {
            return modint(this->m_value*(-1));
        }
        
        modint& operator+=(const modint& t){
            return ((*this) = modint(t.m_value + this->m_value));
        }
        modint& operator-=(const modint& t){
            return ((*this) += (-t));
        }
        modint& operator*=(const modint& t){
            return ((*this) = modint(this->m_value * t.m_value));
        }
        modint& operator/=(const modint& t){
            return ((*this)*=t.inv());
        }
        
        friend modint operator+(modint a , modint b){
            return a+=b;
        }
        friend modint operator-(modint a , modint b){
            return a-=b;
        }
        friend modint operator*(modint a , modint b){
            return a*=b;
        }
        friend modint operator/(modint a , modint b){
            return a/=b;
        }
        friend bool operator==(const modint& a , const modint& b){
            return a.m_value == b.m_value;
        }
        friend bool operator!=(const modint& a , const modint& b){
            return !(a==b);
        }

        // 順序も定義するだけしておく (min/max や set/map に入れる用)
        friend partial_ordering operator<=>(modint a , modint b){
            if(a.m_value == b.m_value)return partial_ordering::equivalent;
            else if(a.m_value < b.m_value)return partial_ordering::less;
            return partial_ordering::greater;
        }

        // 出力
        friend ostream& operator<<(ostream& os, const modint& x){
            os << x.m_value;
            return os;
        }

        // 入力
        friend istream& operator>>(istream& is, modint& x){
            long long value;
            is >> value;
            x = modint(value);
            return is;
        }

        // キャスト演算
        template<typename IntType>
        explicit operator IntType(){
            return IntType(this->m_value);
        }

        /*
            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
    };




    /*
        nCr % MOD の計算
        計算量はO(r) . TODO : modfrac を作る
    */
    template<long long MOD>
    modint<MOD> modcomb(long long n , long long r){
        auto N = modint<MOD>(1);
        auto K = modint<MOD>(1);
        for(int i = 0 ; i < r ; i++){
            N *= modint<MOD>(n-i);
            K *= modint<MOD>(i+1);
        }
        return N/K;
    }

    /*
        log_a{b} mod MOD を計算する。
        - つまり、a^x == b mod MOD なる最小の非負整数 x を計算する。
            |- このような x は、存在するなら MOD 未満である (a の冪乗の周期より)
        - 存在しなければ RE とするので、存在判定は上位スコープで行う。
        離散対数問題と呼ばれます

        Copyright © (c) okoteiyu 2024. 
        Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
    */
    template<long long MOD>
    long long modlog(long long a , long long b){
        long long t = sqrt(MOD);
        vector B(t,modint<MOD>(0));
        B[0] = b;
        unordered_map<long long,priority_queue<long long>> find;
        find[int(B[0])].push(0);
        auto a_inv = modint<MOD>(a).inv(); 
        for(int i = 1 ; i < B.size() ; i++ ){
            B[i] = B[i-1]*a_inv;
            find[int(B[i])].push(-i);
        }
        long long rlim = MOD/t;
        auto apt = modint<MOD>(1);
        auto at = modint<MOD>(a).pow(t);
        long long ans = MOD;
        for(int p = 0 ; p <= rlim ; p++ ){
            if(!find[int(apt)].empty()){
                long long q = find[int(apt)].top()*(-1);
                ans = min(p*t+q,ans);
            }
            apt *= at;
        }
        assert(0 <= ans && ans < MOD);
        return ans;
    }


    /* 
        素数 mod で高速なシグマ計算を行う。
        combination のための階乗テーブルを計算するので、指数の上限 K に対して前計算 O(Klog(MOD)) 時間
        - 上限 K が決まっていて、前計算 O(Klog(MOD)) , クエリ O(1) の combination の代わりとしても使える
        - Σ(i≦n) i^k の計算が O(k^2) 時間 (n に依存しない)
    */
    template<long long MOD>
    class FastSigma{
        using modint = modint<MOD>;
        /*   
            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
        // [k][n] := Σ(i≦n) i^k が計算ずみかどうか
        unordered_map<long long ,unordered_map<long long , bool>> m_ready;
        // [k][n] := Σ(i≦n) i^k の値
        unordered_map<long long ,unordered_map<long long , modint>> memo;
        // 階乗 , 階乗の逆数 , 累乗 , 逆数 のテーブルを前計算
        vector<modint> fac , fac_inv , inv;
        public:
        FastSigma(){}
        // 指数の上限を与える (計算のために少し余分に必要)
        FastSigma(long long k_max)
            :fac(k_max+3),fac_inv(k_max+3),inv(k_max+3){
            fac[0] = modint(1);
            for(long long k = 1 ; k < fac.size() ; k++){
                fac[k] = fac[k-1]*k;
            }
            for(long long k = 0 ; k < fac.size() ; k++){
                fac_inv[k] = fac[k].inv();
                if(k >= 1)inv[k] = fac_inv[k] * fac[k-1];
            }
        }
        // nCr の計算 (変数名がこんがらがらないように注意)
        modint nCr(const long long n ,const long long r){
            assert(0 <= n && n < fac.size());
            assert(0 <= r && r <= n);
            return (fac[n]*fac_inv[r])*fac_inv[n-r];
        }
        // Σ(i≦n) i^k を計算
        modint F(const long long n ,const long long k){
            assert(1<=n);
            assert(0 <= k && k < fac.size());
            if(k == 0)return n;
            if(m_ready[n][k])return memo[n][k];
            auto res = modint(n+1).pow(k + 1) - modint(1);
            for(long long p = 0 ; p <= k + 1 - 2 ; p++){
                res -= this->nCr(k + 1,p)*this->F(n,p);
            }
            memo[n][k] = res*inv[k+1];
            m_ready[n][k] = true;
            return memo[n][k];
        }
        /*   
            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
    };






    /*
        Z/modZ での高速フーリエ変換。mod の値を変えると、色々変える必要がある。
        |
        - MOD = 998244353 = 119×2^23 + 1 の場合
            |- A = 119;
            |- power = 23; 
            |- base = 3;
        - この場合、2^power までの長さに対応できる。MODの値による
    */
    template<long long mod_fft, long long A_fft, long long base_fft, long long power_fft, typename = enable_if_t<bool(A_fft*(1ll<<power_fft) + 1 == mod_fft)>>
    class FFTCalclator{
        /*   
            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
        private:
        using mint = modint<mod_fft>;
        vector<vector<mint>> g;// g[i][x] := 1 の 2^i 乗根 の x 乗

        // 数列 a を Z/modZで フーリエ変換(数論変換する), inv = true なら逆変換
        // 再帰で実装しているので、逆変換後にサイズで割る操作は外で行う!!!!! (めっちゃ注意)
        vector<mint> fft_sub(vector<mint> a , bool inv) const& {
            if(a.size() == 1)return a;
            int N = 1; 
            int bitsize = 0;
            // 2 の冪乗の形になるように整形
            while(N < a.size()){
                N <<= 1;
                bitsize++;
            }
            a.resize(N,0);
            vector<mint> tmp1,tmp2;
            for(int i = 0 ; i*2 < N ; i++){
                tmp1.push_back(a[2*i]);
                tmp2.push_back(a[2*i+1]);
            }
            tmp1 = fft_sub(tmp1,inv);
            tmp2 = fft_sub(tmp2,inv);
            
            for(int i = 0 ; i < N ; i++){
                mint w = g[bitsize][i];
                if(inv)w = g[bitsize][N-i];
                if(i < tmp1.size())a[i] = tmp1[i] + tmp2[i]*w;
                else a[i] = tmp1[i-tmp1.size()] + tmp2[i-tmp1.size()]*w;
            }
            return a;
        }
        
        public:
        FFTCalclator(){
            // 1 の 2^i乗根 gi を計算し、g[i][x] = gi^x も前計算
            g = vector<vector<mint>>(power_fft);
            mint g_power = base_fft;
            g_power = g_power.pow(A_fft*2);
            for(int i = int(g.size())-1 ; i >= 0 ; i--){
                // 0 乗根 から 2^i 乗根 (inclusive) までのテーブルを作る
                g[i].resize((1<<i)+1);
                mint now = 1;
                for(int x = 0 ; x < int(g[i].size()) ; x++){
                    g[i][x] = now;
                    now*=g_power;
                }
                g_power *= g_power;
            }
        }
        FFTCalclator(const FFTCalclator&) = delete;
        FFTCalclator(FFTCalclator&&) = delete;
        FFTCalclator& operator=(const FFTCalclator&) = delete;
        FFTCalclator& operator=(FFTCalclator&&) = delete;

        // 通常の fft 
        vector<mint> FFT(vector<mint>a) const {
            return fft_sub(a,false);
        }
        // fft の逆変換
        vector<mint> IFFT(vector<mint>a) const {
            a = fft_sub(a,true);
            // 実装の設計上、サイズで割る操作は fft_sub の外で行う
            for(int i = 0 ; i < a.size() ; i++)a[i] /= mint(a.size());
            return a;
        }

        // a,bを畳み込み　　-->  c[t] = Σ a[i]*b[t-i] .. (0<=i<=t) となるような数列：c を計算する
        // long long とかは mint に暗黙にキャストされるので, long long でも OK
        vector<mint> convolution(vector<mint> a ,vector<mint> b) const {
            const int sz = int(a.size()) + int(b.size()) - 1;//畳み込みの結果のサイズ
            int t = 1;//計算に使うサイズに整形
            while(t < sz)t <<= 1;
            a.resize(t,mint(0));
            b.resize(t,mint(0));
            vector<mint> A,B;
            vector<mint> C(t);
            
            A = FFT(a);
            B = FFT(b);
            for(int i = 0 ; i < t ; i++)C[i] = A[i]*B[i];
            return IFFT(C);// 逆変換
        }
        /*   
            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
    };




    /*
        - 形式的冪級数ライブラリ : 自身のメンバを変化させるので、const 参照はできない。
        - x^i の係数を A[i] と置く。この式は A.size()-1 次式である。
            |- 係数は mod 付き整数
            |- 整数を定数項のみの多項式にキャストして扱うことで、実装の共通化をしている
                |- mod 付き整数 (剰余環) なので、0 以外の整数 (定数項のみの多項式) 
                   / に対して必ず逆数が存在するような場合がある (mod998244353の場合など)
                   / この時、定数項のみの多項式同士で「%」演算をしても、欲しい結果は得られない !!!!!! 
                   / 「%」演算はあくまで多項式の徐算である。係数は剰余環であることを忘れないでね
        - 多項式同士の演算ができる。
        |
        - 係数を変化させたりすると次数も変化する可能性がある (最高次係数を 0 にした場合、次数が減る)。
            |- 適切なタイミングで format 関数を呼び出して A を整形する
                |- format() := A.back()が 0 であるかぎり、pop_back する。ただし、A[0] = 0の場合はpopしない
                |- FPS の実装を含むメンバ関数の冒頭と、値を返す前に必ず呼ぶ。
            |- バグをなくすため、基本的にオブジェクトが呼び出される全てのタイミングで format() を呼び出すようにする
                |- 基本的に、format() 後の A に準拠するようにしてください
            |- [i] 演算子は x^i の次数のコピーを返す
            |- set(i,d) := x^i の係数を d にする 
                |- 0 を set すると format が発生するかもしれないのでかなり注意
            |- cin_and_set(i) := 入力から d を受け取り、x^i の係数を d にする 
        |
        - 次数の上限 deg_max を 設定する!!!
            |- 次数が deg_max を超える部分は切り捨てる。
            |- 余分に設定しておく方が良いが、遅い場合は set_max_deg で必要な分だけ確保する。
            |- deg_max を小さくした場合、それ以前に計算した deg_max より大きい項は 0 になるので注意
            |- 新たに FPS を生成するときも、set_deg を忘れないで
        |
        - 注意
            |- 整数も定数項のみの FPS (多項式) として扱う。
            |- オペレータは FPS と整数の掛け算,割り算に FFT,ダブリング を使うため、整数との演算では for 文を回すプログラムを書く方が良い
    */
    template<long long mod,long long A_fft ,long long base_fft ,long long power_fft>
    class FPS {

        /*   
            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
        private:
        inline static const FFTCalclator<mod, A_fft, base_fft, power_fft> calculator;
        using mint = modint<mod>;

        public:
        // 関連クラスを定義
        // 疎な多項式 : (次数,係数) のペア列
        typedef vector<pair<long long, mint> > SparseFPS;
        
        protected:
        /*----------------------------------------------------------------------------------------------------------------
            メンバ
        ----------------------------------------------------------------------------------------------------------------*/

        // [i] := x^i の項の係数
        mutable vector<mint> A;

        // 次数の上限。FFT の上限も考慮しながら必要な分だけ動的に set_max_deg(v) で確保する
        int deg_max = 2000000;


        /*----------------------------------------------------------------------------------------------------------------
            内部処理用関数
        ----------------------------------------------------------------------------------------------------------------*/

        /*
            A の係数を書き換えると、次数も変化する可能性があるので、正確な次数が必要な場合のために A を整形する。
            format() := A.back() が 0 であるか、次数上限より大きいかぎり pop_back する。ただし A[0] は pop しない  
        */
        void format() const& {
            while(this->A.size() > 1 && this->A.back()==0){
                this->A.pop_back();
            }
            while(this->A.size() > deg_max+1){
                this->A.pop_back();
            }
            return;
        }

        // 先頭 s 項のコピーを取り出す (*this mod x^s を返す)
        FPS prefix(int s) const& {
            FPS res((*this));    
            res.set_max_deg(deg_max);
            s = max(s,1);
            res.A.resize(s,mint(0));
            return res;
        }

        // 先頭 s 項を左右反転した多項式のコピーを取り出す
        FPS rev(int s) const& {
            FPS res((*this));
            res.set_max_deg(deg_max);
            res.A.resize(s,mint(0));
            reverse(res.A.begin(),res.A.end());
            return res;
        }

        // mod x^k でのthisの逆元を求める。(this*g = 1 mod x^k である g を求める)
        FPS inv(int k) const& {
            if((*this)[0] == 0)assert(0);
            FPS res((*this)[0].inv());
            res.set_max_deg(deg_max);
            for(int i = 1; i < k ; i *=2 ){
                res = res*2 - ((this->prefix(i*2)*res).prefix(i*2)*res).prefix(i*2);
            }
            return res.prefix(k);
        }

        public:
        /*----------------------------------------------------------------------------------------------------------------
            コンストラクタなど
        ----------------------------------------------------------------------------------------------------------------*/

        FPS() : A(1,mint(0)){}
        FPS(const vector<mint>&& A_){
            this->A = A_;
            this->format();
        }
        
        FPS(const FPS& g){
            g.format();
            this->deg_max = g.deg_max;
            this->A = g.A;
        }

        FPS(FPS&& g){
            swap(this->deg_max, g.deg_max);
            this->A.swap(g.A);
        }

        // コピー&スワップ イディオム (後述の、整数やvectorから格上げされた FPS も受け取る)
        FPS& operator=(FPS g){
            swap(this->deg_max, g.deg_max);
            this->A.swap(g.A);
            return *this;
        }

        // 整数系クラスからのキャスト (整数から格上げされた modint も受け取る)
        FPS(mint p) : FPS(vector<mint>(1,p)) {} 

        
        // 次数の最大値を設定する。
        void set_max_deg(int mx){
            deg_max = mx;
            this->format();
        }

        //この式が何次式かを返す(formatされた N を返す)
        int degree() const& {
            this->format();
            return int(this->A.size())-1;
        }

        //Aが確保している領域(format済み)
        int size() const& {
            this->format();
            return int(this->A.size());
        }

        // 次数の上限
        int degree_limit() const& {
            return deg_max;
        }

        // 入力から x^i の係数を受け取る
        void cin_and_set(int i){
            long long d;
            cin >> d;
            this->set(i,d);
        }

        /*----------------------------------------------------------------------------------------------------------------
            出力やアクセスなど
        ----------------------------------------------------------------------------------------------------------------*/
        //出力
        friend std::ostream& operator<<(std::ostream& stream, const FPS& F){
            stream << F.degree() <<"次式 : " ;
            for(int i = 0 ; i < F.size() ; i++){
                stream << F[i] << "x^"<<i ;
                if(i < F.size()-1)stream << " + ";
            }
            return stream;
        }
                
        // x^i の係数を read-only で取得
        const mint operator [](int i) const& {
            // 多項式の次数より大きい範囲は 0 で埋まっている
            if(i >= this->A.size())return mint(0);
            return this->A[i];
        }

        // x^i の係数を d にする (mint は int などから暗黙キャスト可能)
        void set(int i , mint d){
            // 元と同じなら無視する。
            if(d == (*this)[i])return;
            if(i >= this->A.size())this->A.resize(i+1, mint(0));
            this->A[i] = d;
            this->format();
        }


        /*----------------------------------------------------------------------------------------------------------------
            FPS クラス自身を用いた計算
        ----------------------------------------------------------------------------------------------------------------*/

        // F^-1 の次数制限ギリギリまでを取得
        FPS inv() const& {
            this->format();
            return inv(deg_max+1);
        }

        FPS operator-() const& {
            return (*this)*(-1);
        }

        /*----------------------------------------------------------------------------------------------------------------
            FPS と整数系クラスのオペレーター
            注意 : FPS も mint も整数から変換コンストラクト可能だが、オーバーロードは解決の優先度によっていい感じに解決される。
        ----------------------------------------------------------------------------------------------------------------*/

        // 自身に x^p をかける (シフト)
        FPS& operator<<=(uint32_t p){    
            this->format();
            vector<mint> nA(p + this->A.size(), 0);
            for(int i = p ; i < nA.size() ; i++)nA[i] = this->A[i-p];
            this->A.swap(nA);
            this->format();
            assert(this->degree() <= this->degree_limit());
            return *this;
        }

        // 自身を x^p でわり、あまりは切り捨て
        FPS& operator>>=(uint32_t p){
            this->format();
            assert(this->A.size() - p >= 0);
            vector<mint> nA(this->A.size() - p, 0);
            for(int i = 0 ; i < this->A.size() ; i++){
                if(i - p < 0)continue;
                nA[i-p] = this->A[i];
            }
            this->A = nA;
            this->format();
            return *this;
        }

        // 定数を加算する
        FPS& operator+=(mint c){
            this->format();
            this->A[0] += c;
            return *this;
        }
        // 減算
        FPS& operator-=(mint c){
            return (*this) += (-c);
        }
        // 乗算
        FPS& operator*=(mint c){
            this->format();
            for(int i = 0 ; i <= this->degree() ; i++)this->A[i] *= c;
            this->format();
            return *this;
        }

        // 徐算(割り算)
        FPS& operator/=(mint c){
            this->format();
            for(int i = 0 ; i <= this->degree() ; i++)this->A[i] /= c;
            this->format();
            return *this;
        }
        // 和
        friend FPS operator+(FPS f , mint c){
            return (f+=c);
        }
        friend FPS operator+(mint c, FPS f){
            return (f+=c);
        }
        // 差
        friend FPS operator-(FPS f , mint c){
            return (f-=c);
        }
        friend FPS operator-(mint c, FPS f){
            return (-f+c);
        }
        // 積
        friend FPS operator*(FPS f , mint c){
            return (f*=c);
        }
        friend FPS operator*(mint c, FPS f){
            return (f*=c);
        }
        // 商
        friend FPS operator/(FPS f , mint c){
            return (f/=c);
        }
        friend FPS operator/(mint c, FPS f){
            return (c*f.inv());
        }
        // シフト
        friend FPS operator<<(FPS f , uint32_t c){
            return (f<<=c);
        }
        friend FPS operator>>(FPS f , uint32_t c){
            return (f>>=c);
        }

        /*----------------------------------------------------------------------------------------------------------------
            FPS と FPS のオペレーター
        ----------------------------------------------------------------------------------------------------------------*/

        //Fの係数を自身の係数に加算する
        FPS& operator+=(const FPS& F){
            this->format();
            F.format();
            int deg = min(F.degree() , deg_max);
            for(int i = 0 ; i <= deg ; i++ )this->set(i,(*this)[i] + F[i]);
            this->format();
            return *this;
        }

        // 引き算
        FPS& operator-=(const FPS& F){
            return (*this) += -F;
        }     

        // 自身に多項式 F をかける
        FPS& operator*=(const FPS& F){
            this->format();
            F.format();
            vector<mint> C = calculator.convolution(this->A , F.A);
            std::swap(this->A , C);
            this->format();
            return *this;
        }


        // this を「this = F*G + r」 である G に変更する。this/F にするわけではない点に注意。
        FPS& operator/=(const FPS& F){
            this->format();
            F.format();
            assert(F.size() != 0);
            if(F.size() > (*this).size()){
                while((*this).size()>1)this->A.pop_back();
                this->set(0,mint(0));
                return *this;
            }
            const int k = this->size() - F.size() + 1;
            *this = (this->rev(this->size())*F.rev(F.size()).inv(k)).rev(k);
            return *this;
        }

        // this を「this = F*G + r」 である r に変更する
        FPS& operator%=(const FPS& F){
            this->format();
            F.format();
            assert(F.size() != 0);
            if(F.size() > (*this).size())return *this;
            (*this)-=((*this)/F)*F;
            this->format();
            return *this;
        }
        
        
        
        friend FPS operator+(FPS F , FPS G){
            return F+=G;
        }
        friend FPS operator-(FPS F , FPS G){
            return F-=G;
        }
        friend FPS operator*(FPS F , FPS G){
            return F*=G;
        }
        friend FPS operator/(FPS F , FPS G){
            return F/=G;
        }
        friend FPS operator%(FPS F , FPS G){
            return F%=G;
        }

        friend bool operator==(const FPS& F , const FPS& G){
            F.format();
            G.format();
            if(G.degree() != F.degree())return false;
            for(int i = 0 ; i < F.size();i++)if(G[i] != F[i])return false;
            return true;
        }
        friend bool operator!=(const FPS& F , const FPS& G){
            return !(F==G);
        }

        // 順序も定義だけしておく (次数の大きさ順)
        friend partial_ordering operator <=>(const FPS& F , const FPS& G){
            return F.degree()<=>G.degree();
        }


        /*----------------------------------------------------------------------------------------------------------------
            SparseFPS に関連するオペレーター。全て揃えるのは大変なので必要最低限のみ
        ----------------------------------------------------------------------------------------------------------------*/
        

        // 過疎っている多項式をかける vpll (first : 次数 , second 係数)
        FPS &operator*=(const SparseFPS& F) {
            this->format();
            FPS res(vector<mint>(1,mint(0)));
            for(const auto& y : F){
                FPS tmp(*this);
                tmp <<= y.first;
                res += tmp*y.second;
            }
            *this = res;
            this->format();
            return *this;
        }

        
        // 過疎っている多項式で割る vpll (first : 次数 , second 係数)
        FPS &operator/=(const SparseFPS& F) {
            this->format();
            int n = (*this).size();
            auto p = F.front();
            mint ic = p.second.inv();
            F.erase(F.begin());
            for(int i = 0 ; i < n ; i++ ) {
                for (const auto& y : F) {
                    int j = y.first;
                    mint b = y.second;
                    if (j > i) break;
                    this->set(i,(*this)[i] - (*this)[i-j] * b);
                }
                this->set(i,(*this)[i]*ic);
            }
            this->format();
            return *this;
        }


        friend FPS operator+(FPS F ,SparseFPS G){
            F.format();
            for(const auto& a : G)F.set(a.first,F[a.first]+a.second);
            F.format();
            return F;
        }

        friend FPS operator+(SparseFPS G , const FPS& F){
            return F+G;
        }
        friend FPS operator*(FPS F ,SparseFPS G){
            return F*=G;
        }
        friend FPS operator*(SparseFPS G , const FPS& F){
            return F*G;
        }
        friend FPS operator/(FPS F ,SparseFPS G){
            return F/=G;
        }
        friend FPS operator/(SparseFPS G ,FPS F){
            return G*F.inv();
        }

            Copyright © (c) okoteiyu 2024. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
    };


    

    /* デカい int ( FFT 使用なので 8300000 桁ぐらいまで計算可能 ) */
    struct LargeInt{
        /*   
            Copyright © (c) okoteiyu 2025. 
            Released under the MIT license(https://opensource.org/licenses/mit-license.php) 
        */
        protected:
        // 
        // 表現可能な桁数。(FFT および、その後の format で digit_mod/2 に収まる範囲に設定する。)
        inline static const int digit_limit = 4000500;
        // 多項式の係数と対応させる基数
        inline static const int digit_unit = 10;
        // FFT のための MOD の値
        inline static const int digit_mod = 998244353;

        
        // デカい桁を多項式で管理する
        mutable FPS<digit_mod,119,3,23> F;

        // 負かどうかを管理して遅延評価
        mutable bool sign_negative = false;

        // 各桁について [0,9] からはみ出た部分を上の桁にシフトする。自身が持つ余計な領域を縮小する役割も兼ねる。
        // ++ 結果が負の値になるような format は不可能とする ++
        // 計算結果を返す際は必ず呼ぶようにする。
        void format() const& {
            // F.size() の仕様で、最上位の 0 でない桁まで領域を縮小してくれる。
            // 0 未満または 10 以上の部分を上の桁に配分していく。
            for(int d = 0 ; d < F.size() ; d++){
                if(F[d].val() < digit_unit)continue;
                int D = F[d].val();
                if(F[d].val()*2 > digit_mod)D -= digit_mod;// 負の値を再現 (digit_mod の上半分は負の値とする)
                // 負の分も考慮
                F.set(d,(D%digit_unit+digit_unit)%digit_unit);
                int over_flow = D - (D%digit_unit+digit_unit)%digit_unit;
                over_flow /= digit_unit;
                // 一つ上の桁に足す。ここで、一つ上の桁の値が |digit_mod/2| を超えないことが保証される。
                F.set(d+1 , F[d+1] + over_flow);
            }
            // 0 は負ではない
            if(F.degree() == 0 && F[0] == 0)this->sign_negative = false;
        }

        public:
        // 文字列からのパーサ
        LargeInt(string I){
            // FFT 可能なサイズに抑える (実際にこのサイズを確保するわけではない (FPS を参照))
            F.set_max_deg(digit_limit);
            // 桁を下から見るために反転
            reverse(I.begin(),I.end());
            if(I.back() == '-'){
                sign_negative = true;
                I.pop_back();
            }
            for(int d = 0 ; d < I.size() ; d++){
                assert(isdigit(I[d]));
                F.set(d , (I[d]-'0'));
            }
        }

        LargeInt() : LargeInt(to_string(0)){}
        LargeInt(long long X) : LargeInt(to_string(X)){}

        LargeInt(const LargeInt& L){
            L.format();
            this->F = L.F;
            this->sign_negative = L.sign_negative;
        }
        LargeInt(LargeInt&& L){
            this->F = std::move(L.F);
            this->sign_negative = L.sign_negative;
        }

        LargeInt& operator=(LargeInt L){
            this->F = std::move(L.F);
            this->sign_negative = L.sign_negative;
            return *this;
        }        


        // 桁数
        int digit() const& {
            return F.size();
        }

        // 文字列に変換
        string toString() const& {
            string res = "";
            for(int d = 0 ; d < F.size() ; d++){
                assert(F[d].val() < digit_unit);
                res.push_back('0' + F[d].val());
            }
            reverse(res.begin(),res.end());
            return res;
        }

        // 符号
        LargeInt operator-() const& {
            LargeInt res = *this;
            res.sign_negative ^= true;// 反転
            if(F.degree() == 0 && F[0] == 0)res.sign_negative = false;
            //res.format();
            return res;
        }

        // 絶対値
        LargeInt abs() const& {
            if(this->sign_negative)return -(*this);
            return (*this);
        }

        // 大小関係
        friend partial_ordering operator<=>(const LargeInt& a , const LargeInt& b)  {
            if(a.sign_negative && !b.sign_negative)return partial_ordering::less;
            if(!a.sign_negative && b.sign_negative)return partial_ordering::greater;
            assert(a.sign_negative == b.sign_negative);
            bool negative = a.sign_negative;// 後から符号を見て反転
            partial_ordering res = partial_ordering::unordered;
            if(a.digit() < b.digit()) res = partial_ordering::less;
            else if(a.digit() > b.digit()) res = partial_ordering::greater;
            else {
                string sa = a.toString();
                string sb = b.toString();
                if(sa == sb)return partial_ordering::equivalent;
                res = (sa <=> sb);// 同じサイズなら辞書順に変換
            }
            assert(res == partial_ordering::less || res == partial_ordering::greater);
            // 符号も見る
            if(negative){
                if(res == partial_ordering::less)res = partial_ordering::greater;
                else res = partial_ordering::less;
            }
            return res;
        }

        friend bool operator==(const LargeInt& a , const LargeInt& b)  {
            return bool(a.F == b.F && a.sign_negative == b.sign_negative);
        }
        friend bool operator!=(const LargeInt& a , const LargeInt& b)  {
            return !(a == b);
        }
        
        LargeInt& operator+=(LargeInt I){
            /*
                結果の符号で場合分けする。
                format() の結果が負にならないようにする。
            */
            FPS buffer_fps = this->F;
            if(this->sign_negative)buffer_fps *= -1;

            if(I.sign_negative)buffer_fps -= I.F;
            else buffer_fps += I.F;

            if(this->sign_negative && I.sign_negative){
                this->sign_negative = true;
                this->F = -buffer_fps;
            }else if(this->sign_negative == false && I.sign_negative){
                if((*this).abs() < I.abs()){
                    this->sign_negative = true;
                    this->F = -buffer_fps;
                }else{
                    this->sign_negative = false;
                    this->F = buffer_fps;
                }
            }else if(this->sign_negative && I.sign_negative == false){
                if((*this).abs() > I.abs()){
                    this->sign_negative = true;
                    this->F = -buffer_fps;
                }else{
                    this->sign_negative = false;
                    this->F = buffer_fps;
                }
            }else{
                this->sign_negative = false;
                this->F = buffer_fps;
            }

            this->format();
            return (*this);
        }

        LargeInt& operator-=(const LargeInt& I){
            return (*this) += (-I);
        }

        LargeInt& operator*=(const LargeInt& I){
            if(this->sign_negative != I.sign_negative)this->sign_negative = true;
            else this->sign_negative = false;
            this->F *= I.F;
            this->format();
            return *this;
        }
        // 負の値については、FPS の徐算を定義していないので未定義とする
        LargeInt& operator/=(const LargeInt& I){
            if(this->sign_negative != I.sign_negative)this->sign_negative = true;
            else this->sign_negative = false;
            this->F -= (this->F % I.F);
            this->format();
            this->F /= I.F;
            this->format();
            return *this;
        }       
        // 負の値については、FPS の徐算を定義していないので未定義とする
        LargeInt& operator%=(const LargeInt& I){
            assert(this->sign_negative == false && I.sign_negative == false);
            this->F %= I.F;
            this->format();
            return *this;
        }

        friend LargeInt operator+(LargeInt a ,LargeInt b){
            return (a+=b);
        }
        friend LargeInt operator-(LargeInt a ,LargeInt b){
            return (a-=b);
        }
        friend LargeInt operator*(LargeInt a ,LargeInt b){
            return (a*=b);
        }
        friend LargeInt operator/(LargeInt a ,LargeInt b){
            return (a/=b);
        }
        friend LargeInt operator%(LargeInt a ,LargeInt b){
            return (a%=b);
        }

        // 出力
        friend ostream& operator<<(ostream& os , const LargeInt& a){
            if(a.sign_negative)os << '-';
            // a.F.degree() の仕様で、最上位の 0 でない桁まで領域を縮小してくれる。
            for(int i = a.F.degree() ; i >= 0 ; i--){
                assert(a.F[i].val() < LargeInt::digit_unit);
                os << a.F[i].val();
            }
            return os;
        }
        
        // 入力
        friend istream& operator>>(istream& is , LargeInt& a){
            string s;
            is >> s;
            a = LargeInt(s);
            return is;
        }
