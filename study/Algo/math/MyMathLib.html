    // math
    template<class T> T pow2(T x) {return x*x;}// x の 2 乗
    template<class T> double dist2(pair<T,T> a , pair<T,T> b) {return sqrt<double>(pow2<double>(a.first-b.first) + pow2<double>(a.second-b.second));}// 2 点間の距離
    double toRadian(double D){return D*PI/180.0;}// 度数法 (°) から弧度法 (ラジアン)
    double toDegree(double R){return R*180.0/PI;}// 弧度法 (ラジアン) から度数法 (°) 
    //最小公倍数、最大公約数
    template<class Integer>Integer gcd(Integer x, Integer y) { return (y!=0) ? gcd(y, x % y) : x; }
    template<class Integer>Integer lcm(Integer a, Integer b) { return (a / gcd(a, b)) * b; }
    // 組み合わせ
    template<class Integer>Integer nPr(ll n , ll r){Integer res = 1;assert(n>=r);while(r-->0)res *= Integer(n--);return res;}
    template<class Integer>Integer nCr(ll n,  ll r){// 必ず k <= n 
        if ( r * 2 > n ) r = n - r;
        return nPr<Integer>(n,r) / nPr<Integer>(r,r);
    }

    /*
        拡張ユークリッド : 不定方程式の特殊解を見つける
        a,b に対して、ax+by = gcd(a,b) の解 (x_,y_) を一つ見つける。
        この時、ax + by = c の特殊解は c が g = gcd(a,b) の倍数なら、(x_*c/g , y_*c/g) である。
        c が g の倍数でないなら解なし
    */
    pair<long long , long long> Egcd(long long a, long long b) {
        if (b == 0) {return {1,0};}
        long long g = gcd<long long>(abs(b),abs(a)%abs(b));
        auto&& [y,x] = Egcd(abs(b),abs(a)%abs(b));
        y -= abs(a)/abs(b) * x;
        if(a<0)x*=-1;if(b<0)y*=-1;//符号を合わせる
        return {x,y};
    }

    /* 
        行列ライブラリ  :  h 行 w 列の行列 
        - 必ず明示的にコンストラクタを呼び出して初期化して!!!
            |- 初期化子リストによる初期化では、H,W などのメンバが初期化されないので

    */
    template<class T>
    class Matrix{
        public:
        // 単位行列を返すやつ
        inline static Matrix<T> E(int N) {
            Matrix<T> res(N,N,0);
            for(int i=0; i<N; i++) res[i][i] = T(1);
            return res;
        }
        protected :
        int H , W;
        vector< vector<T> > mat;// Matrix クラスなら各行の長さは必ず同じである保証がある
        public:
        Matrix() {}
        Matrix(int h, int w, T val = T(0)) : H(h),W(w), mat(h, vector<T>(w, val)){}

        // 要素の型を変換
        template<typename C> operator Matrix<C>(){
            Matrix<C> res(this->H,this->W);
            // 中身の型も自作ライブラリの可能性があるので、明示的にキャストする。詳しくは okoteiyu ライブラリ設計規則を参照。
            for(int i = 0 ; i < H ; i++ )for(int j = 0 ; j < W ; j++)res[i][j] = C((*this)[i][j]);
            return res;
        }

        int size_h() const& { return H; }// 行数
        int size_w() const& { return W; }// 列数

        // 上書き可能アクセス。
        vector<T>& operator[](int i) { return mat[i]; }
        // const 引数用
        const vector<T>& operator[](int i) const { return mat[i]; }


        // = オペレータはコンパイラに任せる
        Matrix<T>& operator+=(Matrix<T> x) {
            assert(this->mat.size() == x.size());
            for(int i = 0; i < H; i++) for(int j=0; j < W; j++) mat[i][j] += x[i][j];
            return (*this);
        }

        Matrix<T> operator-(){
            Matrix<T> res = (*this);
            for(int i = 0; i <H; i++)for(int j=0; j < W; j++) res[i][j] *= T(-1);
            return res;
        }

        Matrix<T>& operator-=(Matrix<T> x){return ((*this) += -x);}
        Matrix<T>& operator*=(Matrix<T> B) {
            assert(this->W == B.H);
            Matrix<T> res(this->H,B.W);
            for(int i=0; i < H; i++) for(int j=0; j < B.W; j++ ) for(int k=0; k < B.H; k++) res[i][j] += mat[i][k] * B[k][j];
            (*this) = res;
            return (*this);
        }

        Matrix<T>& operator*=(T x){
            for(int i=0; i < H; i++)for(int j=0; j < W; j++ )mat[i][j]*=x;
            return (*this);
        }

        friend Matrix<T> operator +(Matrix<T> A , Matrix<T> B){return (A+=B);}
        friend Matrix<T> operator -(Matrix<T> A , Matrix<T> B){return (A-=B);}
        friend Matrix<T> operator *(Matrix<T> A , Matrix<T> B){return (A*=B);}
        friend Matrix<T> operator *(Matrix<T> A , T x){return (A*=x);}
        friend Matrix<T> operator *(T x ,Matrix<T> A){return A*x;}

        bool operator==(const Matrix<T>& x){
            if(this->H != x.H)return false;
            if(this->W != x.W)return false;
            for(int i = 0; i < H ; i++)for(int j = 0 ; j < W ; j++)if(this->mat[i][j] != x[i][j])return false;
            return true;
        }
        bool operator!=(const Matrix<T>& x){return bool(!(*this == x));}

        // H == W の場合のみ OK 
        Matrix<T> pow(long long b) const& {
            assert(H==W);
            Matrix<T> res = E(H);
            Matrix<T> a = (*this);
            Matrix<T> acum = a;
            while(b > 0){
                if(b&(1ll))res*=acum;
                acum*=acum;
                b>>=1;
            }
            return res;
        }


        // 自身を LU 分解して [L,U] をペアで返す。second には LU 分解時の行交換の回数を持つ。
        // 自身が階段型の正方行列でなければエラーになる
        // ただし、線型独立でない (下の方に 0 のみの行が存在する) 場合も LU 分解はできる
        pair<pair<Matrix,Matrix>,int> LU_Decomposition(){
            assert(this->size_h() == this->size_w());
            const int N = this->size_h();
            int swapcnt = 0;
            Matrix U = (*this);
            Matrix L = Matrix<T>::E(N);// L は単位行列で初期化
            for(int k = 0 ; k < N ; k++){
                // k 行以降で、k 列目が非 0 の行を探す
                for(int v = k ; v < N ; v++){
                    if(U[v][k] == T(0))continue;
                    if(v == k)break;// 行交換する必要がない場合
                    // 行交換する場合
                    U[k].swap(U[v]);
                    swapcnt++;
                    break;
                }
                for(int i = k + 1 ; i < N ; i++){
                    if(U[i][k] == T(0))continue;
                    T tmp = U[i][k]/U[k][k];// 徐算を減らす
                    L[i][k] = tmp; // L は掃き出し法の徐算で使った係数と一致
                    // U はガウスの消去法の係数行列と一致
                    for(int j = k ; j < N ; j++)U[i][j] -= tmp*U[k][j];
                }
            }
            return {{L,U},swapcnt};
        }

        // 自身の行列式
        // 正方行列でなければエラー
        T det(){
            assert(this->size_h() == this->size_w());
            Matrix A = *this;
            pair<pair<Matrix,Matrix>,int>LUD = A.LU_Decomposition();
            const int N = A.size_h();
            T prod = T(1);
            for(int i = 0 ; i < N ; i++)prod *= LUD.first.second[i][i];
            for(int i = 0 ; i < LUD.second ; i++)prod *= T(-1);
            return prod;
        }

        // 内部実装の vector のコピーを返す。
        vector<vector<T>> table() const& {return this->mat;}

        friend ostream& operator << (ostream& os, const Matrix<T> p){
            os << endl;
            for(int i = 0 ; i < p.size_h() ; i++ ){
                os << " - ";
                for(int j = 0 ; j < p.size_w() ; j++)os << p[i][j] << " ";
                os << endl;
            }
            os << endl;
            return os;
        }
    };


    // A の要素を構築する Xor 基底を返す
    template<typename unsigned_integer>
    vector<unsigned_integer> XorBase(vector<unsigned_integer> A){
        vector<unsigned_integer> base;
        for(unsigned_integer v : A){
            for(unsigned_integer e : base)v = min(v, v ^ e);
            if(v != 0) base.push_back(v);
        }
        return base;
    }

    // 精度を求める場合は __float128 とかにする (遅いし double にキャストしないと出力できない点に注意) 
    using Real = long double;

    // 回転行列 (原点を軸として反時計回りに rad (ラジアン,弧度法) 回転) 
    MyMath::Matrix<Real> rotate_matrix(Real rad){
        MyMath::Matrix<Real> res(2,2,0);
        res[0][0] = cos(rad);
        res[0][1] = sin(rad)*(-1);
        res[1][0] = sin(rad);
        res[1][1] = cos(rad);
        return res;
    }  

    // 度数法での回転行列
    MyMath::Matrix<Real> rotate_matrix_withdeg(Real deg){return rotate_matrix(MyMath::toRadian(deg));}


    /* 
        - T 型平面座標系の点の座標と id(識別番号) を管理するパッケージ。
        - 2 行 1 列の行列とも捉える。
    */
    template<typename T>
    struct Point{
        public:
        // 原点を返す static な機能
        inline static Point<T> Origin() {return Point<T>(0,0);}
        int id = -1;// 点に番号を降りたいときとか (デフォルトは -1 ) 
        T x,y;// 座標
        Point(){}
        Point(T x , T y ):x(x),y(y){}

        // 行列を点に変換 (要素の型も U -> T にキャスト)、明示的キャスト
        template<typename U>explicit Point(MyMath::Matrix<U> A) : x(T(A[0][0])),y(T(A[1][0])) {assert(A.size_h() == 2 && A.size_w() == 1);}

        // 中身の型を変更(キャスト)。中身の型は自作ライブラリの可能性があるので、中身は明示的にキャストする。詳しくは okoteiyu ライブラリ設計規則を参照。
        template<typename C> operator Point<C>() const& {Point<C>res(C(this->x),C(this->y)) ;res.id = this->id;return res;}

        // 点として一致するか (id はみない)
        bool operator ==(Point<T> t){return bool(this->x == t.x && this->y == t.y);}
        bool operator !=(Point<T> t){return !((*this) == t);}
        // 2 × 1 行列にキャストできる (縦の行列) (キャストは明示的)
        template<typename C = T> explicit operator MyMath::Matrix<C>() const& {
            MyMath::Matrix<T> res(2,1);
            res[0][0] = C(this->x);
            res[1][0] = C(this->y);
            return res;
        }

        // (a,b) だけ平行移動
        void slide(T a , T b){
            this->x += a;
            this->y += b;
        }

        // 座標を t 倍にする (負も OK)
        void extend(T t){
            this->x*=t;
            this->y*=t;
        }

        // 原点を軸として反時計回りに rad 回転する(ラジアン) したものを返す (ただし、座標は Real 型になる)
        Point<Real> rotate(Real rad) const& {return Point<Real>(rotate_matrix(rad)*(MyMath::Matrix<Real>(*this)));}
        // 度数法で rotate 
        Point<Real> rotate_deg(Real deg) const& {return this->rotate(MyMath::toRadian(deg));}
        // 出力
        friend ostream& operator << (ostream& os, const Point<T> p){os << "ID = " << p.id << " : (" << p.x << " ," << p.y << "). ";return os;}
    };


    /*  
        - 二次元ベクトル。Point みたいに 2 つの成分を持つ。
        - Point を使わずに、MyVector を位置ベクトルとして使用するのもいいかも。
        - Point と違い、演算が数学的な意味に準拠している。
            |- + は合成、* は内積など
    */
    template<class T>
    struct MyVector{
        T x , y;
        MyVector(){}
        // x,y 成分を渡すコンストラクタは基底クラスの方を使う
        MyVector(T x_,T y_):x(x_),y(y_){}
        // 位置ベクトル
        MyVector(Point<T> A):x(A.x),y(A.y){}
        // 始点と終点を渡す。(A → B) (向きに注意)
        MyVector(Point<T> A  , Point<T> B):x(B.x-A.x),y(B.y-A.y){}
        // 行列をベクトルに変換 (要素の型も U -> T にキャスト)
        template<typename U>
        MyVector(MyMath::Matrix<U> A) : x(T(A[0][0])),y(T(A[1][0])) {assert(A.size_h() == 2 && A.size_w() == 1);}

        // 中身の型を変更(キャスト)。中身の型は自作ライブラリの可能性があるので、中身は明示的にキャストする。詳しくは okoteiyu ライブラリ設計規則を参照。
        template<typename C> operator MyVector<C>() const& {return MyVector<C>(C(this->x),C(this->y));}

        // 一致するか (id はみない)
        bool operator ==(MyVector<T> t){return bool(this->x == t.x && this->y == t.y);}
        bool operator !=(MyVector<T> t){return !((*this) == t);}

        MyVector<T> operator -(){MyVector<T> res = (*this);res.x*=-1;res.y*=-1;return res;}
        MyVector<T>& operator +=(MyVector<T> v){this->x += v.x;this->y += v.y;return (*this);} // 合成
        MyVector<T>& operator -=(MyVector<T> v){(*this) += -v;return (*this);}// 合成
        MyVector<T>& operator *=(T v){this->x *= v;this->y *= v;return (*this);}// 拡大
        // Real でないと丸められてしまう。
        MyVector<T>& operator /=(T v){assert(v!=0); this->x /= v;this->y /= v;return (*this);}// 縮小

        // ベクトル A と B の内積
        friend T operator *(MyVector<T> A , MyVector<T> B) {return A.x*B.x + A.y*B.y;}
        // ベクトルの演算。Point と ベクトルで演算も可能。Point は演算を持たないので、重複することもない。
        friend MyVector<T> operator +(MyVector<T> A , MyVector<T> B){return (A+=B);}
        friend MyVector<T> operator -(MyVector<T> A , MyVector<T> B){return (A-=B);}
        // スカラ倍
        friend MyVector<T> operator *(MyVector<T> A ,T v){return (A*=v);}
        friend MyVector<T> operator *(T v , MyVector<T> A){return (A*=v);}
        friend MyVector<T> operator /(MyVector<T> A ,T v){return (A/=v);}

        // 点や行列へのキャスト (明示的にする必要がある)
        template<typename C = T> explicit operator Point<C>() const& {return Point<C>(C(this->x),C(this->y));}

        template<typename C = T>explicit operator MyMath::Matrix<C>() const& {
            MyMath::Matrix<C> res(2,1);
            res[0][0] = C(this->x);
            res[1][0] = C(this->y);
            return res;
        }

        // 出力
        friend ostream& operator << (ostream& os, const MyVector<T> p){os << " (" << p.x << " ," << p.y << "). ";return os;}
        
        // 始点を軸として反時計回りに rad 回転する(ラジアン) したものを返す (ただし、座標は Real 型になる)
        MyVector<Real> rotate(Real rad) const& {return rotate_matrix(rad)*(MyMath::Matrix<Real>(*this));}
        // 度数法で rotate 
        MyVector<Real> rotate_deg(Real deg) const& {return this->rotate(MyMath::toRadian(deg));}
        // このベクトルの長さを返す (必ず実数型)
        Real size() const& {return sqrt(Real(this->x)*Real(this->x) + Real(this->y)*Real(this->y));}
        // このベクトルの長さの二乗を返す。ルートを取らなくて良いので、整数の場合に誤差が出ない。オーバーフロー注意
        T size_pow2() const& {return this->x*this->x + this->y*this->y;}
        // このベクトルと同じ向きの単位ベクトルを返す。
        MyVector<Real> unit() const& {return MyVector<Real>(*this)/(this->size());}
    };




    // 二点を結ぶ線分。exclusive = true なら端点は含まない
    template<typename T>
    struct Segment{
        Point<T> A,B;// 端点 A-B 
        bool exclusive = false;// 端点を含めるか

        public:
        Segment(){}
        Segment(Point<T> a , Point<T> b , bool exclusive_ = false):A(a),B(b),exclusive(exclusive_){}
        // (a,b) から (c,d) への線分
        Segment(T a,T b,T c,T d, bool exclusive_ = false):A(Point<T>(a,b)),B(Point<T>(c,d)),exclusive(exclusive_){}
        // 線分が同じか
        bool operator ==(Segment<T> t){return bool(this->A == t.A && this->B == t.B && this->exclusive == t.exclusive);}

        // 中身の型を変更(キャスト)。中身の型は自作ライブラリの可能性があるので、中身は明示的にキャストする。詳しくは okoteiyu ライブラリ設計規則を参照。
        template<typename C> operator Segment<C>(){return Segment<C>(Point<C>(this->A),Point<C>(this->B),exclusive);}
        // A -> B のベクトルに変換 (キャストは明示的)
        template<typename C = T>explicit operator MyVector<C>() const& {return MyVector<C>(this->A,this->B);}

        // この線分が x 軸と垂直かどうか (傾きが INF)
        bool vertical() const& {return bool((this->A.x - this->B.x) == T(0));}
        // 傾き (計算する前に、垂直かを調べないといけない)
        template<typename C> C slope() const& {assert(!(this->vertical()));return C(A.y-B.y)/C(A.x-B.x);}

        // 線分が点 p をカバーするか
        // 線分の型に依存する
        // ちょっと怖いので、バグってたらここかも
        bool cover(Point<T> p) const& {
            Point<T> L = A;
            Point<T> R = B;
            // x,y 座標それぞれに対して、p が線分の座標の範囲内かを調べる
            if(L.x > R.x)std::swap(L,R);
            if(this->exclusive && !(bool(L.x < p.x && p.x < R.x)))return false;
            else if(!this->exclusive && !(bool(L.x <= p.x && p.x <= R.x)))return false;

            if(L.y > R.y)std::swap(L,R);
            if(this->exclusive && !(bool(L.y < p.y && p.y < R.y)))return false;
            else if(!this->exclusive && !(bool(L.y <= p.y && p.y <= R.y)))return false;

            // AB が p を通るなら Ap と pB の傾きが同じはず
            return bool((p.y-A.y)*(B.x-p.x) == (B.y-p.y)*(p.x-A.x));
        }


        /*  
            P かつ Q である座標、つまり交点を求める。
            P , Q が交点を持つなら、true とその交点の Point を返し、持たないなら false と未定義の Point を返す (線分の型に依存)
            **** 傾きが同じ場合、交差するかの旨だけ報告して、second は未定義 ****
        */
        friend pair<bool,Point<T>> operator &(Segment<T> P ,Segment<T> Q){
            // 三角形 ABC の面積の 2 倍 (誤差をなくすため)
            auto area_doubled_ = [&](Point<T> A , Point<T> B , Point<T> C) -> T {
                MyVector<T> vB(A,B),vC(A,C);
                T ans = ((vB.x)*(vC.y) - (vB.y)*(vC.x));
                if(ans<0)ans*=T(-1);
                return ans;
            };
            // 傾きが同じ場合
            if((P.B.y-P.A.y)*(Q.B.x-Q.A.x) == (Q.B.y-Q.A.y)*(P.B.x-P.A.x)){
                if(P.exclusive||Q.exclusive){// exclusive は、{false,false} 以外は全部 {true,true} にして判定して OK
                    P.exclusive = true;
                    Q.exclusive = true;
                }
                return {bool(P.cover(Q.A) || P.cover(Q.B) || Q.cover(P.A) || Q.cover(P.B)),Point<T>::Origin()};
            }

            /* P,Q の交点が存在するかは、Q.B が以下の平行四辺形 OSRU の内部 (or 周上) に含まれるかどうかを見て判定可能 */
            // OSRU は、P の 2 つの端点に Q.A が重なるように Q を平行移動してできる平行四辺形 (実際に Q を平行移動するわけではない)
            Point<T> O = P.A;
            Point<T> S = P.B;
            Point<T> U = Point<T>(O + MyVector<T>(Q.A,Q.B));// 位置ベクトル (P.A) + (Q.A->Q.B) 
            Point<T> R = Point<T>(S + MyVector<T>(Q.A,Q.B));// 位置ベクトル (P.B) + (Q.A->Q.B) 

            // 平行四辺形 OSRU の面積
            T OSRU = area_doubled_(O,S,R) + area_doubled_(O,U,R);

            // 点 p_ が平行四辺形 ORSU の内部または周上に含まれるかどうかを調べる
            // 周上に含まれてはいけない場合を除いて考える (面積が 0 かどうかで判定可能)
            if(P.exclusive)if(area_doubled_(Q.B,S,R) == 0|| area_doubled_(Q.B,U,O) == 0)return {false,Point<T>::Origin()};
            if(Q.exclusive)if(area_doubled_(Q.B,O,S) == 0|| area_doubled_(Q.B,R,U) == 0)return {false,Point<T>::Origin()};
            // 面積の和で内外を確かめる
            if(OSRU != area_doubled_(Q.B,O,S) + area_doubled_(Q.B,S,R) + area_doubled_(Q.B,R,U)+ area_doubled_(Q.B,U,O))return {false,Point<T>::Origin()};

            // 交点の位置ベクトルを O + OS ベクトルの合成で求める。OS ベクトルの長さは、面積の比から特定する。
            return {true , Point<T>(MyVector<T>(O) + (MyVector<T>(O,S)*area_doubled_(O,Q.A,Q.B))/(area_doubled_(O,Q.A,Q.B) + area_doubled_(S,Q.A,Q.B)))};
        }
        // & 演算と同じ
        friend pair<bool,Point<T>> operator &&(Segment<T> P ,Segment<T> Q){return P&Q;}

        // 出力
        friend ostream& operator << (ostream& os, const Segment<T> p){
            os << " (" << p.A.x << " ," << p.A.y << ") -> " << "(" << p.B.x << " ," << p.B.y << ") : ";
            if(p.exclusive) cout << "exclusive" << endl;
            else cout << "inclusive" << endl;
            return os;
        }
        
    };


    /*
        面積関係
        面積関係
        面積関係
    */

    /*
        符号付き面積を 2 倍したもの。▲ABC の符号付き面積は、
            - A から見て B→C (の小さい方の角) が反時計回りなら正の面積となる。
            - A から見て B→C (の小さい方の角) が時計回りなら負の面積となる。
        +++ A,B,C が整数座標なら返り値も整数となる。 +++
    */ 
    template<typename Z = Real>
    Z signed_area_doubled(Point<Z> A , Point<Z> B , Point<Z> C){
        MyVector<Z> vB(A,B),vC(A,C);
        return ((vB.x)*(vC.y) - (vB.y)*(vC.x));
    }

    // ▲ABC の符号付き面積。signed_area_doubled を参照
    template<typename T = Real>
    T signed_area(Point<T> A , Point<T> B , Point<T> C){return signed_area_doubled<T>(A,B,C)/T(2);}


    // 多角形の面積 × 2 を求める (座標が整数座標の多角形なら、整数を返すことができる。)
    // 多角形の外周の点を反時計回りに並べたものを P とし、点の重複/自己交叉 はないものとする。
    template<typename Z = Real>
    Z Area_Doubled(vector<Point<Z>> P){
        Z res = 0;// P[0] ,P[i] ,P[i+1] の三角形の符号付き面積を足していく。(凸じゃない多角形にも対応している)
        for(int i = 1 ; i < int(P.size()) - 1; i++)res += signed_area_doubled<Z>(P[0],P[i],P[i+1]);
        return res;
    }
    // 多角形の外周の点を反時計回りに並べたものを P とし、点の重複/自己交叉 はないものとする。
    template<typename T = Real>
    T Area(vector<Point<T>> P){return Area_Doubled<T>(P)/T(2);}

        点の関係
        点の関係
        点の関係
    */


    /* 
        点 A がどの象限にあるかを返す
        - 第 1 象限 : 0 ≦ y , 0 < x の領域
        - 第 2 象限 : 0 < y , x ≦ 0 の領域
        - 第 3 象限 : y ≦ 0 , x < 0 の領域
        - 第 4 象限 : y < 0 , 0 ≦ x の領域
    */
    template<typename T>
    int Quadrant(Point<T> A){
        MyVector<T> p(Point<T>::Origin(),A);
        if(0 <= p.y && 0 < p.x )return 1;
        if(0 < p.y && p.x <= 0)return 2;
        if(p.y <= 0 && p.x < 0)return 3;
        return 4;
    }

    // 2 つのベクトル A , B のなす角 (ラジアン) を返す。
    // 鋭角にしたければ PI からひく
    template<typename T = Real>T angle(MyVector<T> A, MyVector<T> B){return acos(T(A*B)/T(A.size()*B.size()));}

    // 角 a-b-c (b の角) の角度のうち小さい方 (ラジアン) を返す
    template<typename T = Real>T angle(Point<T> a, Point<T> b, Point<T> c) {return angle<T>(MyVector<T>(b,a),MyVector<T>(b,c));}

    // centerから見て、a → b (の小さい方の角) が++++反時計回り++++かどうか(includeがtrueなら、同じ角度でも反時計回りとみなす)
    // なす角が小さい方で判定する
    // なす角が 180° の場合、include が真なら true を返し、そうでないなら false を返す
    template<typename T>
    bool ccw(Point<T> Center , Point<T> a , Point<T> b , bool include = false){
        T S = signed_area_doubled(Center,a,b);
        if(include)return S >= 0;
        return S > 0;
    }
    // (cx,cy) からみて 点(x1,y1) -> 点(x2,y2) (の小さい方の角) が反時計回りかどうか(includeがtrueなら、同じ角度でも反時計回りとみなす。)
    template<typename T>
    bool ccw(T cx, T cy ,T x1 , T y1 , T x2 , T y2 , bool include = false){return ccw( Point<T>(cx,cy) , Point<T>(x1,y1) , Point<T>(x2,y2) , include);}

    // 点 a,b,c が同一直線上かどうか (座標が実数だと精度がアレなので注意)
    template<typename T>// ccw に同一角度を含める場合とそうでない場合の結果から計算
    bool same_line(Point<T> a , Point<T> b , Point<T> c){
        if(a == b || b == c || c == a)return true;
        return bool(signed_area_doubled(a,b,c) == 0);
    }

    // 点 (x1,y1) ,(x2,y2) ,(x3,y3) が同一直線上かどうか
    template<typename T>
    bool same_line(T x1 , T y1 , T x2 , T y2 ,T x3, T y3 ){return same_line(Point(x1,y1) , Point(x2,y2) ,Point(x3,y3));}


    // Point のリストから重複を弾き、全ての要素が unique になったものを返す
    // 副作用 : 座標 (x , y) の辞書順でソートしたものを返す
    template<typename T>
    vector<Point<T>> uniquePointList(vector<Point<T>> P){
        if(P.size()<=1)return P;
        // (x,y) の辞書順でソート
        sort(P.begin(),P.end(),[&](Point<T>a , Point<T> b)->bool {if(a.x == b.x)return a.y < b.y;return a.x < b.x;});
        vector<Point<T>> res;
        res.push_back(P.front());
        for(int i = 1 ; i < P.size(); i++)if(P[i] != P[i-1])res.push_back(P[i]);
        return res;
    }
    

    /*
        点群の左下 → 右上をカバーする凸な点群 (左下 → 右上の順にグラハムスキャンする) (ccw , same_line に依存)
        :    \____|  のように選ばれることはあるが、|______/ のようにはならないことに注意
        :    |______/ ではなく  ______/  のようになる (左上が選ばれない)
        必要に応じてみる順番を変えることができるが、ccw などの記述も変える際はバグに注意
        |
        - include_online = true なら、凸包の周上にある点を全て含める (3 点が同一直線上にある場合など)
        - ==== P に重複要素があってはならない ====
        - ==== P に重複要素があってはならない ====
        - ==== P に重複要素があってはならない ====
    */
    template<typename T>
    vector<Point<T>> GrahamScan(vector<Point<T>> P , bool include_on_line = false){
        vector<Point<T>> CH;
        // x 座標の < で比較。同じ場合は y 座標の < で比較 :  正順は 左下 → 右上  , 逆順は 右上 → 左下 
        sort(P.begin(),P.end(),[&](Point<T> a ,Point<T> b){
            if(a.x == b.x)return a.y < b.y;
            return a.x < b.x;
        });
        if(P.size() <= 2)return P;
        int cur = 0;
        while(CH.size()<2)CH.push_back(P[cur++]);
        for( ; cur < P.size() ; cur++){
            while(CH.size()>=2){
                // x 座標が直近の順に CH[|CH|-2] , CH.back()  , P[cur] となっている
                bool pop_ = false;
                // 3 点が同一直線上 か CH.back() からみた角度が ccw なら CH.pop_back();
                if(same_line(P[cur] , CH.back() , CH[int(CH.size())-2]) && include_on_line == false)pop_ = true;
                if(ccw(CH.back() , CH[int(CH.size())-2] , P[cur]))pop_ = true;
                if(pop_)CH.pop_back();
                else break;
            }
            CH.push_back(P[cur]);
        }
        return CH;
    }

    // - 点群の上下をグラハムスキャンして凸包を求める (凸包の左下の点から反時計回りに一周する順番で返す)
    // - include_online = true なら、凸包の周上にある点を全て含める (3 点が同一直線上にある場合など)
    template<typename T>
    vector<Point<T>> ConvexHull(vector<Point<T>> P , bool include_on_line = false){
        P = uniquePointList(P);// 重複を弾く
        if(P.size()<=2)return GrahamScan(P,include_on_line);// 小さい場合、ソートだけして返す
        vector<Point<T>> CH;
        CH = GrahamScan(P,include_on_line);// 凸包の左下 → 右上をカバーする凸な点群
        for(Point<T>& p : P){p.x*=-1;p.y*=-1;}// 符号を反転すると元の P の右上からスキャンすることに対応
        vector<Point<T>> tmp = GrahamScan(P,include_on_line);// 凸包の右上 → 左下をカバーする凸な点群
        for(Point<T>& p : tmp){p.x*=-1;p.y*=-1;}
        // 下側と上側では、左下と右上以外の点が重複することはない
        CH.pop_back();
        tmp.pop_back();
        for(Point<T> p : tmp)CH.push_back(p);
        return CH;
    }

    /*
        実数の点群 P から、異なる点同士で最も遠いペアを見つける。
        凸包を作って円環上を調べる。キャリパー法
    */
    template<typename T>
    pair<Point<T>,Point<T>> pairOfFurthestPoints(vector<Point<T>> P){
        P = uniquePointList(P);// 重複を弾く
        vector<Point<T>> CH = ConvexHull(P);// 凸包を作って、尺取り法をする
        assert(CH.size() >= 2);
        /*
            いつか書く
        */
        return {CH.front(),CH.front()};
    }   



    /*  
        実数の点群 P から、異なる点同士で最も近いペアを見つける。
        副作用 : P を **y** 座標でソートする
        最初に一度点群の重複を取り除き、ついでに x 座標でソートする (必要なければしない)。
        |
        - 分割する index を mid_index とし、index で左右に分ける。
            |- また境界の x 座標を mid_x とする。ただし、分割はあくまで index で行う
        - 左右の領域それぞれから、暫定的な最小距離 d を再帰的に計算。
        - 左右の領域の点群を、y 座標の昇順になるようにマージソートのように前から見ながらマージする。
            |- これを P とすることで、P が y 座標でソートされた
            |- P は参照渡し
        - あとは x 座標が mid_x から d 以内の点にのみ注目する。
            |- y 座標の差も同様に d 以内のペアのみを見る。
            |- このペアはかなり少ない
        - 分割統治しながらオマケでマージソートしているので、計算量は O(|P|log|P|)
        - 距離の誤差が怖い場合は、二乗の値で比較する。
            |- オーバーフロー注意
    */
    template<typename T>
    pair<Point<T>,Point<T>> pairOfNearestPoints(vector<Point<T>>& P){
        // 必要に応じて一度だけ x 座標でソート & 重複を弾く
        bool unique_and_sorted = true;
        for(int i = 0 ; i < int(P.size())-1 ; i++){
            if(P[i].x > P[i+1].x)unique_and_sorted = false;// ソートされていない
            if(P[i].x == P[i+1].x && P[i].y == P[i+1].y)unique_and_sorted = false;// 重複あり
        }
        if(!unique_and_sorted)P = uniquePointList(P);
        assert(P.size() >= 2);
        Real d = MyVector<T>(P.front(),P.back()).size();// 2 点の距離の最小値を計算したい。(比較は実数型)
        pair<Point<T>,Point<T>> ans = make_pair(P.front(),P.back());
        if(P.size() == 2){
            if(P[0].y > P[1].y)swap(P[0],P[1]);// y 座標でソート
            return make_pair(P[0],P[1]);
        }
        vector<Point<T>> L,R;
        int mid_index = int(P.size())/2;
        T mid_x = P[mid_index].x;
        // x 座標で分割統治。L,R も (x,y) の辞書順でソートされている
        for(int i = 0 ; i < P.size() ; i++){
            if(i < mid_index)L.push_back(P[i]);// 座標の大小ではなく、index で分割 (半分にするため)
            else R.push_back(P[i]);
        }
        if(L.size()>=2){// L 側を調べる。ついでに L は y 座標でソートされる。
            pair<Point<T>,Point<T>> resL = pairOfNearestPoints(L);
            if(d > MyVector<T>(resL.first,resL.second).size()){
                d = MyVector<T>(resL.first,resL.second).size();
                ans = resL;
            }
        }
        if(R.size()>=2){// R 側を調べる。ついでに R は y 座標でソートされる。
            pair<Point<T>,Point<T>> resR = pairOfNearestPoints(R);
            if(d > MyVector<T>(resR.first,resR.second).size()){
                d = MyVector<T>(resR.first,resR.second).size();
                ans = resR;
            }
        }
        P.clear();// L,R から y 座標でマージソートしながら P に入れ直す (L,R は再帰的に y 座標でソート済みである)
        int lcur = 0;// L の何番目を見てるか
        int rcur = 0;// R の何番目を見てるか
        while(lcur < L.size() || rcur < R.size()){
            if(lcur >= L.size())P.push_back(R[rcur++]);// R のみ
            else if(rcur >= R.size())P.push_back(L[lcur++]);// L のみ
            else if(L[lcur].y < R[rcur].y)P.push_back(L[lcur++]);// L < R
            else P.push_back(R[rcur++]);
        }
        // 点のうち、mid_x と x 座標の差が d 以内のもの同士にのみ注目
        deque<Point<T>> recent;// i (今見ているindex) から直近定数個 (最低 6 個) をメモしておく。
        for(int i = 0 ; i < P.size() ; i++){
            if(abs<Real>(P[i].x - mid_x) > d)continue;
            for(int j = 0 ; j < recent.size() ; j++){
                if(d > MyVector<T>(P[i],recent[j]).size()){
                    d = MyVector<T>(P[i],recent[j]).size();
                    ans = make_pair(P[i],recent[j]);
                }
            }
            recent.push_back(P[i]);
            if(recent.size()>=10)recent.pop_front();
        }
        return ans;
    }
