C++ 言語機能等トピック一覧

内容

C++ の言語機能などを断片的に紹介する記事です。


トピック一覧

  1. 言語機能
    1. constexpr 関数は実引数がコンパイル時定数である場合に限り、返り値がコンパイル時定数になる。返り値がコンパイル時定数になるなら、テンプレート引数リストの中でその値を使用することができる。
    2. ビットフィールドは実質的に参照不可能である。
    3. (TODO : 厳密かつ簡潔に) テンプレートの特殊化で、候補である複数の (シグネチャが一致する) テンプレートから \(1\) つの実体化するテンプレートを解決(特定)する場合を考える。SFINAE という言語機能によって、ある候補の実体化が失敗しても即座にエラーとはならず、別の候補のテンプレートから解決しようとする。全ての候補で解決できなかった時にコンパイルエラーとなる。SFINAE を利用することで、コンパイル時に条件分岐を実現できる。条件分岐とは複数条件を 1 つずつ順に試す手続きなので、その点でも一致する。

#include <utility>

// 終了状態
template<typename T, typename X>
void mapping_sub(T (*f)(X) , vector<T>& buf){
    return;
}

// Tail の先頭要素に f を適用して buf に記録
template<typename T, typename X, typename... Tail>
void mapping_sub(T (*f)(X), vector<T>& buf, X&& head, Tail&&... tail){
    mapping_sub<T,Tail...>(f, buf, std::forward<Tail>(tail)...);
    buf.push_back(f(head));
}

// 引数全てに f を適用し、結果を vector に格納して返す
template<typename T, typename X, typename... Args>
vector<T> mapping(T (*f)(X), Args&&... args){
    vector<T> res;
    mapping_sub(f, res, std::forward<Args>(args)...);
    reverse(res.begin(), res.end());
    return res;
}

// イテレータ付きコンテナに対するマッピング
template<typename T, typename X, typename Container,
    typename = void_t<decltype(declval<Container>().begin())>,
    typename = void_t<decltype(declval<Container>().end())>
>
vector<T> mapping(T (*f)(X), Container C){
    vector<T> res;
    auto itr = C.begin();
    const auto e = C.end();
    while(itr != e)res.push_back( f(*(itr++)) );
    return res;
}


// イテレータ付きコンテナに対する出力
template<typename Container,

    const auto e = C.end();
    while(itr != e)cout << *(itr++) << " ";

std::pair に [] オペレータでアクセスする 

#include <iostream>
#include <cassert>
using namespace std;
template<typename t1 , typename t2>
struct mpair{
    protected:
    pair<t1,t2> m_pr;
    public:
    mpair(){}
    explicit mpair(pair<t1,t2> pr_) : m_pr(pr_){}
    mpair(t1 first , t2 second) : m_pr(first,second) {}
    explicit operator pair<t1,t2>(){return m_pr;}
    /*
        [0] -> t1
        0 は nullptr_t にキャスト可能なので、もう一つの operator[] よりも優先度を上げるとこっちが呼ばれる
    */
    t1& operator[](nullptr_t i){
        return m_pr.first;
    }
    // 自作クラスを使って優先度を下げる。
    struct _dummy_{
        _dummy_(int x){
            assert(x == 1);
        }
    };
    // [1] -> t2 
    t2& operator[](_dummy_ i)  {
        return m_pr.second;
    }
    friend bool operator==(const mpair<t1,t2>& a , const mpair<t1,t2>& b){return a.m_pr == b.m_pr;}
    friend bool operator!=(const mpair<t1,t2>& a , const mpair<t1,t2>& b){return !(a==b);}
    friend partial_ordering operator<=>(const mpair<t1,t2>& a , const mpair<t1,t2>& b){return a.m_pr <=> b.m_pr;}
};

    mpair<string,int> pr;
